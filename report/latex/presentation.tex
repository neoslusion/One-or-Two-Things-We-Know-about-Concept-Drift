\section{Background: The MMD Drift Signal}
% #############################################################################

\begin{frame}{Maximum Mean Discrepancy (MMD)}
	\begin{columns}[T]
		\begin{column}{0.55\textwidth}
			\textbf{Idea:} compare two distributions $P, Q$ via their mean embeddings in an RKHS.
			\begin{equation*}
				\text{MMD}(P, Q) = \sup_{f \in \mathcal{F}} \left| \mathbb{E}_{X \sim P}[f(X)] - \mathbb{E}_{Y \sim Q}[f(Y)] \right|
			\end{equation*}

			\textbf{Empirical estimator} (Gaussian RBF kernel):
			\vspace{-0.5em}
			\begin{align*}
				\widehat{\text{MMD}}^2 & = \underbrace{\tfrac{1}{n^2}\textstyle\sum_{i,j} k(x_i, x_j)}_{\text{within } P}
				+ \underbrace{\tfrac{1}{m^2}\textstyle\sum_{p,q} k(y_p, y_q)}_{\text{within } Q}                                                       \\
				                       & \quad - \underbrace{\tfrac{2}{nm}\textstyle\sum_{i,p} k(x_i, y_p)}_{\text{cross } P{-}Q}
			\end{align*}

			\vspace{-0.3em}
			$k(x,y) = \exp\!\left(-\dfrac{\|x - y\|^2}{2\sigma_k^2}\right)$ \quad (Gaussian RBF)
		\end{column}
		\begin{column}{0.42\textwidth}
			\begin{alertblock}{Role in drift detection}
				\begin{itemize}
					\item $\widehat{\text{MMD}}^2 \approx 0$: distributions look the same.
					\item $\widehat{\text{MMD}}^2 \gg 0$: distributions disagree $\to$ \textbf{drift candidate}.
				\end{itemize}
			\end{alertblock}

			\vspace{0.3em}
			\begin{exampleblock}{\small Why it suits this problem}
				\small
				Each sample is mapped into an RKHS via the kernel; the test compares the mean embeddings of the two windows.\\
				Result: a single scalar test statistic, no distributional assumption, and no labels required at runtime.
			\end{exampleblock}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Online MMD Estimation via Sliding Window}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{Stream-to-Trace Mapping:}
			\begin{itemize}
				\small
				\item A sliding window of size $2l_1$ moves over the raw data stream.
				\item Split into reference window $W_{\text{ref}}$ (first half) and test window $W_{\text{test}}$ (second half).
				\item MMD is computed between $W_{\text{ref}}$ and $W_{\text{test}}$ at each step.
				\item The result is a continuous MMD trace $\sigma(t)$ tracking distributional changes over time.
			\end{itemize}
			
			\vspace{0.3em}
			\begin{block}{\small Detection Signal}
				\small
				When the window covers stable data, $\sigma(t)$ fluctuates around a small noise floor. When crossing a drift, $\sigma(t)$ rises and forms a peak.
			\end{block}
		\end{column}
		\begin{column}{0.48\textwidth}
			\centering
			\includegraphics[width=0.98\linewidth]{theory_sliding_window.png}
			\captionof{figure}{\scriptsize sliding-window MMD calculation: reference window $W_{\text{ref}}$ and test window $W_{\text{test}}$ are compared to produce one point $\sigma(t)$ on the trace.}
		\end{column}
	\end{columns}
\end{frame}

\section{Layer 1: Detection}
% #############################################################################

\begin{frame}[shrink]{ShapeDD --- Shape-Based Drift Detection}
	\begin{columns}[T]
		\begin{column}{0.55\textwidth}
			\textbf{Triangle Shape Theorem:}

			\vspace{0.1em}
			For a sudden drift at $t = 0$, the MMD trace satisfies
			\begin{equation*}
				\sigma(t) = \underbrace{\| P - Q \|}_{\text{drift severity}} \cdot \underbrace{h_l(t)}_{\text{triangular envelope}},
			\end{equation*}
			with $h_l(t) = \max\left(0, 1 - \frac{|l - t|}{l}\right)$.

			\vspace{0.2em}
			\textbf{Processing pipeline:}
			\begin{enumerate}
				\item Maintain a sliding window of $2 l_1$ samples.
				\item Build the kernel matrix $K$.
				\item Compute MMD between the two halves.
				\item Test significance with a permutation test.
			\end{enumerate}

			\vspace{0.2em}
			\begin{alertblock}{\footnotesize Bottleneck}
				\scriptsize
				The permutation test re-builds the kernel thousands of times $\to$ heavy compute and the null distribution must still be estimated correctly.
			\end{alertblock}
		\end{column}
		\begin{column}{0.42\textwidth}
			\centering
			\includegraphics[width=0.85\linewidth]{vis_repeated_sudden_SE.png}
			\vspace{-0.3em}
			\captionof{figure}{\scriptsize Sudden drift produces the predicted triangular MMD trace.}

			\begin{exampleblock}{\footnotesize Why a triangle?}
				\scriptsize
				As the sliding window crosses the change point, the proportion of post-drift samples grows linearly $\to$ MMD rises linearly until the window is fully on the new distribution, then falls back symmetrically.
			\end{exampleblock}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Inverse Density-Weighted MMD (IDW-MMD)}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{Limitation of standard MMD (uniform weights):}
			\begin{itemize}
				\small
				\item Dense majority clusters dominate the discrepancy term.
				\item Early drifts that appear at distribution boundaries are washed out by the bulk.
			\end{itemize}

			\vspace{0.5em}
			\begin{exampleblock}{\small Idea: weight by inverse local density}
				\small
				Down-weight points sitting in dense regions ($w_i \downarrow$); amplify isolated points near the boundary ($w_i \uparrow$). Boundary changes that signal drift get more influence on the test statistic.
			\end{exampleblock}
		\end{column}
		\begin{column}{0.48\textwidth}
			\centering
			\includegraphics[width=0.95\linewidth]{theory_idw_weighting.png}
			\captionof{figure}{\scriptsize MMD with uniform weights (A) vs.\ IDW-MMD (B) which automatically compresses the core and amplifies the boundary.}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{IDW-MMD Formulation \& Calibration}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{IDW-MMD Weighting Function:}
			\begin{block}{Inverse Local Density Weights}
				\small
				\begin{equation*}
					w_i \propto \frac{1}{\sqrt{d_i} + \epsilon}, \quad d_i = \sum_{j \neq i} k(x_i, x_j)
				\end{equation*}
				$\epsilon$ is a small constant (e.g., $0.5$) for numerical stability.
			\end{block}

			\textbf{Weighted MMD Estimator:}
			{\scriptsize
			\begin{equation*}
				\widehat{\text{MMD}}^2_{\text{IDW}} = \sum_{i \neq j} \textcolor{univblue}{w_i w_j}\, k(x_i, x_j) + \sum_{p \neq q} \textcolor{univblue}{v_p v_q}\, k(y_p, y_q) - \frac{2}{nm} \sum_{i,p} k(x_i, y_p)
			\end{equation*}}
		\end{column}
		\begin{column}{0.48\textwidth}
			\textbf{Statistical Calibration:}
			\begin{alertblock}{\small Gamma-Null Approximation}
				\small
				The permutation test is replaced by a moment-matched Gamma-null p-value (Gretton et al.\ 2009).\\
				\vspace{0.3em}
				Reuses the computed kernel matrix and estimates moments from a few cheap shuffles.\\
				\vspace{0.3em}
				Result: properly calibrated test ($\alpha = 0.05$) with a massive end-to-end speedup over standard permutation.
			\end{alertblock}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}[shrink]{Worked Example: 3-Point Window}
	A minimal observation window with 3 samples:
    \begin{itemize}
        \item Majority core: $x_1, x_2$ sit close together (normal behaviour).
        \item Boundary point: $x_3$ is isolated near the feature boundary (drift candidate).
    \end{itemize}

	\vspace{1em}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{1. Local density $d_i = \sum_j k(x_i, x_j)$}
            \begin{itemize}
                \item Core: $d_1 = k(x_1,x_1) + k(x_1,x_2) \approx 1.9$
                \item Core: $d_2 = k(x_2,x_1) + k(x_2,x_2) \approx 1.9$
                \item Boundary: $d_3 = k(x_3,x_3) + \text{small} \approx \mathbf{1.0}$
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{2. Inverse weights $w_i = 1/\sqrt{d_i}$}
            \begin{itemize}
                \item Core: $w_1 = w_2 = 1/\sqrt{1.9} \approx \mathbf{0.72}$ (down-weighted).
                \item Boundary: $w_3 = 1/\sqrt{1.0} = \mathbf{1.0}$ (kept at full weight).
            \end{itemize}
        \end{column}
    \end{columns}

	\vspace{1em}
	\begin{alertblock}{\small Take-away}
	    \small Standard MMD treats every point equally ($w_i = 1$). IDW gives the boundary point ($w_3 = 1.0$) more weight than the bulk ($w_{1,2} = 0.72$), so a small change at the boundary moves the test statistic. This is what gives early sensitivity to drift while keeping false positives low on stationary data.
	\end{alertblock}
\end{frame}

\section{Layer 2: Classification}
% #############################################################################

\begin{frame}{CDT-MSW --- Supervised Classification Baseline}
	\begin{columns}[T]
		\begin{column}{0.55\textwidth}
			\textbf{CDT-MSW} (Guo et al., 2022) groups drift into:
			\begin{itemize}
				\item \textbf{TCD (Transient):} Sudden, Blip, Recurrent.
				\item \textbf{PCD (Progressive):} Gradual, Incremental.
			\end{itemize}

			\vspace{0.5em}
			\textbf{Growth Process:}
			\begin{itemize}
				\item Tracks the variance of accuracy degradation over a multi-sliding window.
				\item Short degradation $\to$ TCD; sustained degradation $\to$ PCD.
			\end{itemize}

			\vspace{0.5em}
			\begin{alertblock}{Limitation}
				CDT-MSW needs \textbf{ground-truth labels} at every step to compute the accuracy curve --- impractical for IIoT streams where labels arrive late or never.
			\end{alertblock}
		\end{column}
		\begin{column}{0.42\textwidth}
			\centering
			\includegraphics[width=\linewidth]{image/cdt_msw.png}
			\captionof{figure}{\scriptsize CDT-MSW architecture (label-dependent).}

			\vspace{0.5em}
			\begin{exampleblock}{\small Motivation for SE-CDT}
				\small
				Replace the accuracy curve with the \emph{shape} of the MMD signal $\sigma(t)$. The same TCD/PCD distinction can then be made \textbf{without any labels at runtime}.
			\end{exampleblock}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Unsupervised Shape Analysis of $\sigma(t)$}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{How can we type drift without ground-truth labels?}

			Read the \textit{shape} of the MMD signal $\sigma(t)$ around the detection point. SE-CDT extracts \textbf{9 statistical features}:

			\vspace{0.5em}
			\begin{itemize}
				\item \textbf{Geometric (peak structure):}
				    \begin{itemize}
    				    \item $n_p$, WR, CV, SNR: count, width, periodicity, prominence.
    				    \item PPR, DPAR: peak proximity and dual-peak amplitude ratio.
    				\end{itemize}
				\item \textbf{Temporal (behaviour over time):}
				    \begin{itemize}
				        \item LTS, SDS, MS: trend strength, step score, monotonicity.
				    \end{itemize}
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			\centering
			\includegraphics[width=0.98\linewidth]{theory_peak_anatomy.png}
			\captionof{figure}{\scriptsize MMD peak features (e.g., FWHM, Prominence) extracted from the shape of the signal $\sigma(t)$.}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{SE-CDT Classifier: Worked Example}
	\textbf{Trace of one online detection event:}
	
	\vspace{1em}
	Layer 1 has just fired a drift alert at time $t^*$ on the streaming data.
	
	\vspace{1em}
	\begin{enumerate}
	    \item \textbf{Extract Signal Window:} Read the MMD trace $\sigma(t)$ over a window of $2 l_1$ samples around $t^*$.
	    \item \textbf{Calculate Geometric Features:}
	        \begin{itemize}
	            \item Count local maxima $\to$ $n_p = 1$ ($\leq 3$).
	            \item Width at half maximum $\to$ $\text{WR} = 0.12$ ($< 0.15$).
	            \item Peak vs baseline noise $\to$ $\text{SNR} = 3.0$ ($> 2$).
	        \end{itemize}
	    \item \textbf{Evaluate Decision Logic:}
	        \begin{itemize}
	            \item The first rule matches: $n_p \leq 3 \ \land \ \text{WR} < 0.15 \ \land \ \text{SNR} > 2$.
	        \end{itemize}
	\end{enumerate}

	\vspace{1em}
	\begin{alertblock}{Decision}
		$\Rightarrow$ Label = \textbf{Sudden Drift}, which is then dispatched to Layer 3 (Adaptation) to trigger a full model reset.
	\end{alertblock}
\end{frame}

\begin{frame}[shrink]{Hybrid Pipeline: IDW-MMD to Detect, Standard MMD to Classify}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{Why use two MMD variants instead of one?}

			\vspace{0.5em}
			\begin{itemize}
				\item \textbf{IDW-MMD} is sharper and almost never fires on stationary windows --- great as a gatekeeper, but its weighting also \textit{flattens} the slow rise of a Gradual drift, removing the very shape SE-CDT needs.
				\item \textbf{Standard MMD} runs slower but \textit{preserves the natural geometry} of $\sigma(t)$, including plateaus and trends --- ideal as input to the shape classifier.
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{block}{Pipeline}
				\begin{itemize}
					\item \textbf{Layer 1 --- Detection:} \textcolor{univblue}{IDW-MMD} runs continuously as the gatekeeper.
					\item \textbf{Layer 2 --- Classification:} \textcolor{univblue}{Standard MMD} is computed \textit{only at the detection time} $t^*$ to feed SE-CDT.
				\end{itemize}
				$\to$ Cheap to run continuously, accurate where it matters.
			\end{block}

			\vspace{0.1em}
			\centering
			\includegraphics[width=0.75\linewidth]{vis_repeated_gradual_SE.png}
			\captionof{figure}{\scriptsize Same Gradual drift: IDW (bottom) flattens, Standard (top) preserves the slope.}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}[noframenumbering]{Backup: IDW-MMD vs Optimally-Weighted MMD (Bharti et al.)}
	\textbf{Question:} ``How is IDW-MMD different from the Optimally-Weighted MMD of Bharti et al. (2023)?''

	\vspace{0.5em}
	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\begin{block}{Bharti et al. (OW-MMD)}
				\begin{itemize}
					\item Domain: \textbf{likelihood-free inference} (ABC).
					\item Weights chosen by optimising against a target posterior.
					\item Requires solving a constrained convex optimisation per call.
					\item Not designed for streaming drift detection.
				\end{itemize}
			\end{block}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{IDW-MMD (this work)}
				\begin{itemize}
					\item Domain: \textbf{drift detection} in fast data streams.
					\item Weights from a closed-form heuristic: $w_i \propto 1/\sqrt{d_i + \epsilon}$.
					\item Cost: $O(n^2)$ for the kernel matrix, $O(n)$ extra for the weights --- no optimisation step.
					\item Same \textit{weighted MMD} family, different objective and very different runtime.
				\end{itemize}
			\end{exampleblock}
		\end{column}
	\end{columns}

	\vspace{0.5em}
	\textbf{Summary:} both are weighted MMD variants, but they target different problems (ABC vs streaming drift), use different weight functions, and have very different runtime profiles.
\end{frame}

% --- Backup 3: Concept Memory mechanism ---
\begin{frame}[noframenumbering,shrink]{Backup: Why does Concept Memory matter for Recurrent drift?}
	\textbf{Question:} ``Recurrent drift requires concept matching --- how does Concept Memory provide it?''

	\vspace{0.5em}
	\begin{columns}[T]
		\begin{column}{0.52\textwidth}
			\textbf{The detection challenge:}
			\begin{itemize}
				\item A Recurrent stream is a sequence of \textit{repeated} concept switches; each switch produces an MMD peak that, in isolation, looks like a Sudden drift.
				\item Without memory of past concepts, repeated visits are scored independently $\to$ Recurrent collapses into the Sudden branch.
			\end{itemize}

			\vspace{0.5em}
			\textbf{How Concept Memory resolves it:}
			\begin{enumerate}
				\item During every \textit{stable period}, take a snapshot of the current data window and store it in a ring buffer of size $K = 8$.
				\item When a new drift fires, compute Standard MMD between the new post-drift window and every snapshot.
				\item If the smallest MMD $< \mu = 0.15$ $\to$ override the rule output to \textbf{Recurrent}.
			\end{enumerate}
		\end{column}
		\begin{column}{0.45\textwidth}
			\begin{exampleblock}{Why $\mu = 0.15$?}
				\small
				$\mu = 0.05$ falls inside the MMD noise floor (false matches between independent stationary windows). An empirical sweep on H0 calibration data shows $\mu = 0.15$ separates ``revisited concept'' from ``new concept'' reliably.
			\end{exampleblock}

			\vspace{0.3em}
			\begin{alertblock}{Remaining limitations}
				\small
				Two failure modes remain on the harder Recurrent variants:
				\begin{itemize}
					\item Recurrent events that are \textit{too far apart} push the original snapshot out of the ring buffer.
					\item Repeated Gradual streams saturate after a few events, so the post-drift window matches a stored snapshot and is labelled Recurrent rather than Gradual.
				\end{itemize}
				$\to$ Future work: learned threshold and longer memory.
			\end{alertblock}
		\end{column}
	\end{columns}
\end{frame}

			\vspace{0.3em}
			\begin{exampleblock}{Possible mitigations (future work)}
				\begin{itemize}
					\item Run IDW (precision channel) and Standard MMD (recall channel) in parallel and combine the two test statistics.

% --- Backup 7: Permutation vs Gamma null ---
\begin{frame}[noframenumbering]{Backup: Permutation test vs Gamma-null p-value}
	\textbf{Question:} ``Removing the permutation test --- is the Gamma-null p-value still trustworthy?''

	\vspace{0.5em}
	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\begin{alertblock}{Permutation test (original ShapeDD)}
				\begin{itemize}
					\item Reshuffles indices many times per window.
					\item Empirical null, no distributional assumption.
					\item Each shuffle re-builds the kernel matrix $\to$ heavy compute.
				\end{itemize}
			\end{alertblock}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{Gamma-null p-value (IDW-MMD)}
				\begin{itemize}
					\item Moment-matched Gamma fit to the null MMD distribution (Gretton et al.\ 2009).
					\item Estimates the first two moments from a small number of index shuffles, reusing the kernel matrix.
					\item Complexity $O(n^2)$ for the kernel plus a few cheap shuffles.
				\end{itemize}
			\end{exampleblock}
		\end{column}
	\end{columns}

	\vspace{0.5em}

End-to-end runtime is slightly faster than the ShapeDD baseline.

% --- Backup 12: Why the slow subtypes are low ---
\begin{frame}[noframenumbering,shrink]{Backup: Why are Gradual (27.2\%) and Incremental (30.3\%) low?}
	\textbf{Question:} ``The two slow types are the weak point --- what exactly goes wrong?''

	\vspace{0.5em}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{Two systematic confusions (not random):}
			\begin{enumerate}
				\small
				\item \textbf{Gradual $\leftrightarrow$ Incremental overlap on Variance Ratio.} In theory high VR $\Rightarrow$ Gradual (mixture inflates variance), low VR $\Rightarrow$ Incremental (single shift). In practice both land in the ambiguous band, so $\sim$47\% of Incremental events are read as Gradual.
				\item \textbf{Saturation $\to$ Recurrent.} In repeated slow-drift streams $P(X)$ stops changing after a few events; the post-drift window matches a stored snapshot, so Concept Memory labels it Recurrent --- $\sim$57\% of Gradual events.
			\end{enumerate}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{\small Why CAT is unaffected}
				\small
				Both confusions stay \emph{inside} PCD, so the TCD/PCD decision (CAT $=80.1\%$) is correct --- and CAT is what the adaptation layer consumes. The cost of confusing Gradual with Incremental is small (same update family).
			\end{exampleblock}

			\vspace{0.3em}
			\begin{alertblock}{\small Fix (future work)}
				\scriptsize
				Replace the fixed VR thresholds + geometric fallback with a learned classifier on $\sigma(t)$, and separate the Concept-Memory match step from the shape classifier.
			\end{alertblock}
		\end{column}
	\end{columns}
\end{frame}

% --- Backup 13: Adaptation selection bias ---
\begin{frame}[noframenumbering,shrink]{Backup: ``Is the adaptation gain Sudden-biased?''}
	\textbf{Question:} ``Type-Specific wins +17pp --- but the classifier is only $\sim$55\% on subtypes. Does the gain depend on the easy scenarios?''

	\vspace{0.5em}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{Honest answer --- yes, partly:}
			\begin{itemize}
				\small
				\item The three adaptation scenarios (Stepping, Sudden, Mixed) are all \textbf{Sudden-dominated}, exactly where the classifier is strongest (Sudden 82.4\%).
				\item So routing is correct \emph{most of the time in this region}, and any reset on a true sudden drift helps regardless.
				\item We have \textbf{not} tested adaptation on Gradual/Incremental-heavy streams, where subtype accuracy drops to 27--30\%.
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{\small Why the gain still holds}
				\small
				The category decision (TCD vs PCD, 80.1\%) drives the adaptation policy, not the exact subtype. TCD$\to$reset and PCD$\to$fine-tune is the right call even when the subtype within PCD is wrong --- the operational cost of confusing Gradual with Incremental is small (same update family).
			\end{exampleblock}

			\vspace{0.3em}
			\begin{alertblock}{\small Scope claim}
				\scriptsize
				We claim the adaptation benefit \emph{for the sudden/recurrent regime}, not universally. Gradual/Incremental-heavy adaptation is listed as future work (Ch.\ 6).
			\end{alertblock}
		\end{column}
	\end{columns}
\end{frame}
