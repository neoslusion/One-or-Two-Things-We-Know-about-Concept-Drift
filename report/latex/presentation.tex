\section{Theoretical Background}

% --- Slide 5: MMD ---
\begin{frame}{Maximum Mean Discrepancy (MMD)}
	\begin{columns}[T]
		\begin{column}{0.55\textwidth}
			\textbf{Core Concept:} Compare two distributions $P, Q$ in a high-dimensional kernel space:
			\begin{equation*}
				\text{MMD}(P, Q) = \sup_{f \in \mathcal{F}} \left| \mathbb{E}_{X \sim P}[f(X)] - \mathbb{E}_{Y \sim Q}[f(Y)] \right|
			\end{equation*}

			\textbf{Empirical Estimator} (RKHS via Gaussian RBF):
			\vspace{-0.5em}
			\begin{align*}
				\widehat{\text{MMD}}^2 & = \underbrace{\frac{1}{n^2}\sum_{i,j} k(x_i, x_j)}_{\text{Internal } P}
				+ \underbrace{\frac{1}{m^2}\sum_{p,q} k(y_p, y_q)}_{\text{Internal } Q}                                                        \\
				                       & \quad - \underbrace{\frac{2}{nm}\sum_{i,p} k(x_i, y_p)}_{\text{Cross } P \leftrightarrow Q}
			\end{align*}

			\vspace{-0.3em}
			$k(x,y) = \exp\left(-\frac{\|x - y\|^2}{2\sigma_k^2}\right)$ \quad (Gaussian RBF)
		\end{column}
		\begin{column}{0.42\textwidth}
			\begin{alertblock}{Role in Drift Detection}
				\begin{itemize}
					\item $\text{MMD}^2 \approx 0$: Identical distributions.
					\item $\text{MMD}^2 > 0$: Divergence $\to$ \textbf{Drift!}
				\end{itemize}
			\end{alertblock}

			\vspace{0.3em}
			\begin{exampleblock}{\small RKHS Embeddings Intuition}
				\small
				Instead of doing exhaustive parametric modeling, MMD directly maps sample data into a Reproducing Kernel Hilbert Space (RKHS).\\
				The distance between their \textbf{Mean Embeddings} acts as a robust discrepancy metric. This fundamentally enables \textbf{non-parametric} and \textbf{unsupervised} real-time validation.
			\end{exampleblock}
		\end{column}
	\end{columns}
\end{frame}

% --- Slide 6: ShapeDD ---
\begin{frame}[shrink]{ShapeDD --- Shape-Based Drift Detection}
	\begin{columns}[T]
		\begin{column}{0.55\textwidth}
			\textbf{Triangle Shape Theorem:}

			\vspace{0.3em}
			Given a sudden drift at $t=0$, the MMD drift magnitude $\sigma(t)$ forms:
			\begin{equation*}
				\sigma(t) = \underbrace{\| P - Q \|}_{\text{Drift Severity}} \cdot \underbrace{h_l(t)}_{\text{Triangle Shape}}
			\end{equation*}
			where $h_l(t) = \max\left(0, 1 - \frac{|l - t|}{l}\right)$

			\vspace{0.5em}
			\textbf{Processing Pipeline:}
			\begin{enumerate}
				\item Buffer data (Sliding window $2l_1$).
				\item Construct kernel matrix $K$.
				\item Calculate MMD between window halves.
				\item Verify significance via \textbf{Permutation Test}.
			\end{enumerate}
		\end{column}
		\begin{column}{0.42\textwidth}
			\centering
			\includegraphics[width=\linewidth]{vis_repeated_sudden_SE.png}
			\captionof{figure}{\scriptsize Distinctive MMD triangle during Sudden Drift}

			\vspace{0.5em}
			\begin{exampleblock}{\small Geometric Property}
				\small
				As the sliding window steps over the crossing boundary (drift point) of two distinct distributions, the MMD magnitude $\sigma(t)$ inherently plots a \textbf{signature triangle}.\\
				Peak magnitude arises when both samples perfectly center the intersection.
			\end{exampleblock}

			\vspace{0.1em}
			\begin{alertblock}{\small Scalability Bottleneck}
				\small
				The resampling Permutation Test demands \textbf{2,500 iterations per step} $\to$ Unacceptable CPU footprint, capping throughput at \textbf{8,000 samples/s}.
			\end{alertblock}
		\end{column}
	\end{columns}
\end{frame}

			\begin{alertblock}{Limitations of CDT-MSW}

% --- Slide 9: IDW-MMD ---
\begin{frame}[shrink]{Contribution 1: Inverse Density-Weighted MMD Algorithm}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{Limitations of Original MMD (Uniform weights):}
			\begin{itemize}
				\item Uniform weighting allows the dense \textbf{majority clusters} at the center to entirely dominate the statistical discrepancy computations.
				\item Consequence: Drift often originates as outliers at the \textbf{distribution boundary} $\to$ Weak early drift signals are temporarily overshadowed.
			\end{itemize}

			\vspace{0.5em}
			\begin{exampleblock}{\small Signal Amplification Idea}
				\small
				The Inverse Density-Weighted (IDW) mechanism allocates weights inversely proportional to local spatial density.\\
				Dense center points are suppressed ($w_i \downarrow$). Conversely, anomalous divergences at the margins are \textbf{heavily amplified} ($w_i \uparrow$), yielding a starkly sensitive detection signal without sacrificing robustness.
			\end{exampleblock}
		\end{column}
		\begin{column}{0.48\textwidth}
			\textbf{IDW-MMD Solution:} Assign weights $w_i$ relative to local density $d_i$.

			\begin{block}{Restructured MMD Formulation}
				\small
				\begin{equation*}
					w_i \propto \frac{1}{\sqrt{d_i} + \epsilon} \quad \text{where } d_i = \sum_{j} k(x_i, x_j)
				\end{equation*}

				\begin{align*}
					\widehat{\text{MMD}}^2_{\text{IDW}} & = \sum_{i,j} \textcolor{alertred}{w_i w_j} k(x_i, x_j) \\
					                                    & + \sum_{p,q} \textcolor{alertred}{v_p v_q} k(y_p, y_q) \\
					                                    & - \frac{2}{nm} \sum_{i,p} k(x_i, y_p)
				\end{align*}
			\end{block}

			\vspace{0.3em}
			\textbf{Mathematical Elegance:} Shifted from naive iterative sampling (Permutation Test $O(N_{perm} n^2)$) to evaluating the \textbf{Asymptotic p-value} via Gamma approximations ($O(n^2)$). Eliminates 2,500 permutations per timestep!
		\end{column}
	\end{columns}
\end{frame}

% --- Slide 10: IDW-MMD Calculation Example ---
\begin{frame}[shrink]{IDW Weight Decomposition Example (Minimal 3-Point Set)}
	Assume an observation window captures 3 data instances:
    \begin{itemize}
        \item Majority Core: $x_1, x_2$ are clustered closely mapping normal behavior.
        \item Drift Indicator: $x_3$ emerges isolated at the feature boundary.
    \end{itemize}

	\vspace{1em}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{1. Compute local density $d_i = \sum k(x_i, x_j)$}
            \begin{itemize}
                \item Core point: $d_1 = k(x_1, x_1) + k(x_1, x_2) \approx 1.9$
                \item Core point: $d_2 = k(x_2, x_1) + k(x_2, x_2) \approx 1.9$
                \item Boundary point: $d_3 = k(x_3, x_3) + \text{minimal} \approx \textbf{1.0}$
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{2. Extract inverse weights $w_i = \frac{1}{\sqrt{d_i}}$}
            \begin{itemize}
                \item Core weight: $w_1 = w_2 = \frac{1}{\sqrt{1.9}} \approx \textbf{0.72}$ $\to$ Penalized.
                \item Boundary weight: $w_3 = \frac{1}{\sqrt{1.0}} = \textbf{1.0}$ $\to$ Amplified.
            \end{itemize}
        \end{column}
    \end{columns}

	\vspace{1em}
	\begin{alertblock}{\small Theoretical Proof Interpretation}
	    \small Unlike standard MMD ($w_i = 1 \forall i$), IDW \textbf{actively penalizes} signal strength in dense regions ($0.72 < 1.0$), while \textbf{boosting} samples deviating toward the boundary ($1.0 \gg 0.72$). This mechanism leverages spatial distribution shifts, forcing anomalous samples to dominate the discrepancy evaluation, thus identifying subtle concept drift early.
	\end{alertblock}
\end{frame}

% --- Slide 11: SE-CDT Algorithm ---
\begin{frame}[shrink]{Contribution 2: Unsupervised Time-Series Shape Analysis}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{How to classify Drift Types without Ground Truths?}
			
			Instead of verifying accuracy variance, this thesis proposes \textbf{Topological Shape Analysis} by extracting 9 distinct statistical features from $\sigma(t)$:

			\vspace{0.5em}
			\begin{itemize}
				\item \textbf{Geometric Block (Structural Analysis):}
				    \begin{itemize}
    				    \item $n_p$: Total count of local amplitude maxima.
    				    \item $\text{WR}$ (Width Ratio): Pulse spread at High-Half.
    				    \item $\text{SNR}$ (Signal-to-Noise): Peak prominence relative to background noise.
    				\end{itemize}
				\item \textbf{Temporal Block (Behavioral Analysis):}
				    \begin{itemize}
				        \item $\text{LTS}$: Density map residing strictly above the baseline threshold.
				    \end{itemize}
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{\small Rationale: Geometry Correlates with Distribution Dynamics}
				\small
				Different drift architectures inherently output specific energy spectra:
				
				- \textbf{Sudden Drift:} Induces a singular, severe system shock yielding $\textbf{1 isolated peak}$ with extremely narrow bandwidth ($WR \to 0$).\\
				- \textbf{Gradual Drift:} Blended noisy transitions elevate the baseline into a plateau, generally lacking defined peaks.\\
				- \textbf{Blip / Outlier:} Sparks immediate dual-wave transient oscillations ($n_p=2$).
			\end{exampleblock}
			
			\centering
			\includegraphics[width=0.7\linewidth]{vis_mixed_a_SE.png}
			\captionof{figure}{\scriptsize Distinct Sudden Drift geometry ($WR$ narrow, $n_p=1$)}
		\end{column}
	\end{columns}
\end{frame}

% --- Slide 12: SE-CDT Calculation Example ---
\begin{frame}[shrink]{Rule-based Classification Heuristic Example (SE-CDT)}
	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\textbf{Decision Tree Structure (Heuristics):}

			\vspace{0.3em}
			{\small
			\begin{tabular}{@{}p{3.5cm}l@{}}
				\toprule
				\textbf{Verification Conditions}               & \textbf{Classification Result}    \\ \midrule
				$n_p \leq 3, \text{WR}<0.15, \text{SNR}>2$  & \textbf{Sudden Drift}  \\
				$n_p = 2, \text{PPR}<0.2, \text{DPAR}>0.65$ & \textbf{Blip / Outlier}    \\
				$n_p \geq 4, \text{CV}<0.3$             & \textbf{Recurrent Drift}    \\
				$\text{LTS}>0.5 \lor \text{MS}>0.6$  & \textbf{Incremental Drift}   \\
				Fallback (Fails above conditions) & \textbf{Gradual Drift}  \\
				\bottomrule
			\end{tabular}}
		\end{column}
		\begin{column}{0.5\textwidth}
			\textbf{Real-time System Execution Trace:}
			
			Suppose Layer 1 (Detection) just triggered an anomaly threshold violation.
			\begin{enumerate}
			    \item The Vectorizer streams $2l_1$ timestamps surrounding the drift point.
			    \item Counts local maxima: $\textbf{Identifies 1 Peak ($n_p=1 \leq 3$)}$.
			    \item Measures pulse spread at Half-Maximum relative to the window boundary: \textbf{$\text{WR} = 0.12 < 0.15$}.
			    \item Compares against baseline bounds: $\textbf{SNR = 3.0 > 2}$.
			\end{enumerate}
			
			$\Rightarrow$ The sequential feature vector strictly satisfies the first logical branch $\to$ Deterministically labels the event as \textbf{Sudden Drift}.
		\end{column}
	\end{columns}
\end{frame}

% --- Slide 13: Hybrid Design ---
\begin{frame}[shrink]{Hybrid Pipeline: IDW-MMD for Detection, Standard MMD for Classification}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{Decoupled Architecture Motivation:}\\
            Why not utilize a monolithic configuration for both phases?

			\vspace{0.5em}
			\begin{itemize}
				\item \textbf{IDW-MMD (Hyper-sensitive, Noise-canceller):} Inverse weights command unparalleled sensitivity to dimensional boundaries, actively suppressing False Positives. However, it \textbf{flattens} progressive changes (Gradual Drift) $\to$ Sabotages structural data needed for classification.
				\item \textbf{Standard MMD (Low-latency, Geometrically Preserved):} Fast estimations that strictly \textbf{preserve the native geometric slope} of signals $\to$ Provides a flawless pristine input mapping for SE-CDT.
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{block}{System Pipeline Design}
				\begin{itemize}
					\item \textbf{Layer 1 (Detection):} Exclusively runs \textcolor{univblue}{IDW-MMD} $\to$ Acts as an unyielding Gatekeeper, rejecting thousands of stationary epochs, optimizing CPU utilization.
					\item \textbf{Layer 2 (Classification):} Lazily triggers \textcolor{univblue}{Standard MMD} \textbf{only at timestamp $t_{detect}$} $\to$ Locates geometrical insights precisely at the drift incident.
				\end{itemize}
			\end{block}

			\vspace{0.1em}

\begin{frame}[noframenumbering]{Backup: IDW-MMD vs Optimally-Weighted MMD (Bharti et al.)}

\begin{frame}[noframenumbering]{Backup: Permutation Test vs Asymptotic p-value}
	\textbf{Question:} ``By removing the Permutation Test, does the Asymptotic p-value remain accurate?''

	\vspace{0.5em}
	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\begin{alertblock}{Permutation Test (Original ShapeDD)}
				\begin{itemize}
					\item Aggressively shuffles variables 2,500 times.
					\item \textbf{Exact p-value}: Assumes zero underlying statistical distribution bounds.
					\item Complexity: $O(N_{\text{perm}} \cdot n^2)$
					\item Practice: $\sim$6.25 million operations/window.

				\end{itemize}

		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{Asymptotic p-value (IDW-MMD)}
				\begin{itemize}
					\item Operates strictly on \textbf{Asymptotic Gamma approximation}.
					\item Confirmed convergence mapping when block sizes $n \geq 50$ (Current setting: $n = 50$).
					\item Complexity: Analytical $O(n^2)$ (Eigenvalue resolution).
					\item $\to$ Amplified to \textbf{\textasciitilde{}131,000 samples/s}.
				\end{itemize}
			\end{exampleblock}
		\end{column}
	\end{columns}

	\vspace{0.5em}
	\textbf{Metric Verification:} General F1 practically equivalent (0.481), but fundamentally \textbf{collapses structural FP errors to absolute zero} $\to$ Asymptotic approximations technically grant \textbf{enhanced reliability} by drastically lowering variance noise.
\end{frame}
