\section{Cơ sở lý thuyết}

% --- Slide 5: MMD ---
\begin{frame}{Maximum Mean Discrepancy (MMD)}
	\begin{columns}[T]
		\begin{column}{0.55\textwidth}
			\textbf{Ý tưởng:} So sánh 2 phân phối $P$, $Q$ trong không gian kernel cao chiều:
			\begin{equation*}
				\text{MMD}(P, Q) = \sup_{f \in \mathcal{F}} \left| \mathbb{E}_{X \sim P}[f(X)] - \mathbb{E}_{Y \sim Q}[f(Y)] \right|
			\end{equation*}

			\textbf{Ước lượng thực tế} (RKHS với Gaussian RBF kernel):
			\begin{align*}
				\widehat{\text{MMD}}^2 & = \underbrace{\frac{1}{n^2}\sum_{i,j} k(x_i, x_j)}_{\text{Tương đồng trong } P}
				+ \underbrace{\frac{1}{m^2}\sum_{p,q} k(y_p, y_q)}_{\text{Tương đồng trong } Q}                                                        \\
				                       & \quad - \underbrace{\frac{2}{nm}\sum_{i,p} k(x_i, y_p)}_{\text{Tương đồng chéo}}
			\end{align*}

			\vspace{0.3em}
			$k(x,y) = \exp\left(-\frac{\|x - y\|^2}{2\sigma_k^2}\right)$ \quad (Gaussian RBF)
		\end{column}
		\begin{column}{0.42\textwidth}
			\begin{alertblock}{Vai trò trong Drift Detection}
				\begin{itemize}
					\item $\text{MMD}^2 \approx 0$: Hai cửa sổ cùng phân phối
					\item $\text{MMD}^2 > 0$: Phân phối đã thay đổi $\to$ \textbf{Drift!}
				\end{itemize}
			\end{alertblock}

			\vspace{0.5em}
			\begin{exampleblock}{\small Giải thích đơn giản (Ví dụ: So giỏ trái cây)}
				\small
				Làm sao biết "Giỏ Táo" ($P$) và "Giỏ Cam" ($Q$) chứa trái cây khác loại mà \textbf{không cần nếm} (không cần nhãn)?\\
				\textbf{Cách MMD làm:} So sánh chéo \textit{độ tương đồng} về mọi đặc tính (màu, mùi, vỏ). Nếu cả 2 giỏ cùng chứa Táo $\to$ MMD $\approx 0$. Khác loại $\to$ MMD sẽ $> 0$. Không cần biết tên trái cây là gì, khác là phát hiện được!
			\end{exampleblock}
		\end{column}
	\end{columns}
\end{frame}

% --- Slide 6: ShapeDD ---
\begin{frame}{ShapeDD --- Phát hiện drift dựa trên hình dạng tín hiệu}
	\begin{columns}[T]
		\begin{column}{0.55\textwidth}
			\textbf{Định lý Triangle Shape (Kết quả trung tâm):}

			\vspace{0.3em}
			Khi xảy ra drift đột ngột tại $t=0$, tín hiệu drift magnitude:
			\begin{equation*}
				\sigma(t) = \underbrace{\| P - Q \|}_{\text{Độ mạnh drift}} \cdot \underbrace{h_l(t)}_{\text{Hình dạng tam giác}}
			\end{equation*}
			với $h_l(t) = \max\left(0, 1 - \frac{|l - t|}{l}\right)$

			\vspace{0.5em}
			\textbf{4 giai đoạn xử lý:}
			\begin{enumerate}
				\item Thu thập dữ liệu (cửa sổ trượt $2l_1$)
				\item Xây dựng ma trận kernel $K$
				\item Tính MMD giữa 2 nửa cửa sổ
				\item Xác thực bằng \textbf{Permutation Test}
			\end{enumerate}
		\end{column}
		\begin{column}{0.42\textwidth}
			\centering
			\includegraphics[width=\linewidth]{vis_repeated_sudden_SE.png}
			\captionof{figure}{\scriptsize Tín hiệu ShapeDD cho Sudden Drift --- hình tam giác đặc trưng tại điểm drift}

			\vspace{0.5em}
			\begin{exampleblock}{\small Ví dụ: Xe trượt qua "gờ giảm tốc"}
				\small
				Cửa sổ dữ liệu $2l_1$ giống như xe máy băng qua gờ giảm tốc (điểm thay đổi). Bánh trước chạm gờ $\to$ nảy lên. 
				Giữa gờ $\to$ nảy cao nhất (\textbf{đỉnh tam giác}). 
				Bánh sau qua khỏi gờ $\to$ êm lại (\textbf{xuống 0}). ShapeDD chuyên lướt đồ thị đi tìm cái "gờ tam giác" này!
			\end{exampleblock}

			\vspace{0.3em}
			\begin{alertblock}{\small Hạn chế ShapeDD gốc}
				\small
				Permutation Test \textbf{2,500 lần lặp/bước} $\to$ Rất chậm rùa bò ($\sim$8,000 mẫu/s) $\to$ KHÔNG thể chạy real-time!
			\end{alertblock}
		\end{column}
	\end{columns}
\end{frame}

% --- Slide 7: CDT-MSW ---
\begin{frame}{CDT-MSW --- Nền tảng phân loại drift}
	\begin{columns}[T]
		\begin{column}{0.55\textwidth}
			\textbf{CDT-MSW} (Guo et al., 2022): Phân loại drift thành:
			\begin{itemize}
				\item \textbf{TCD} (Transient): Sudden, Blip, Recurrent
				\item \textbf{PCD} (Progressive): Gradual, Incremental
			\end{itemize}

			\vspace{0.5em}
			\textbf{Cơ chế Growth Process:}
			\begin{itemize}
				\item Đo \textit{drift length} qua accuracy variance trên adjoint window
				\item Nếu drift length ngắn $\to$ TCD, dài $\to$ PCD
			\end{itemize}

			\vspace{0.5em}
			\begin{alertblock}{Hạn chế của CDT-MSW}
				\textbf{Yêu cầu labels} để tính accuracy ratio $\to$ Không phù hợp cho môi trường unsupervised (IIoT, streaming)
			\end{alertblock}
		\end{column}
		\begin{column}{0.42\textwidth}
			\centering

			\vspace{0.5em}
			\begin{exampleblock}{\small Động lực cho SE-CDT}
				\small
				Thay accuracy-based growth bằng \textbf{MMD signal analysis} $\to$ hoàn toàn unsupervised
			\end{exampleblock}

% --- Slide 9: IDW-MMD ---
\begin{frame}{Đóng góp 1: Thuyết minh thuật toán Inverse Density-Weighted MMD}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{Vấn đề của MMD nguyên thủy (Trọng số phân bổ đều):}
			\begin{itemize}
				\item Điểm ở \textbf{trung tâm} tụ tập rất đông đúc $\to$ Át hết tiếng nói của người khác.
				\item Drift thực chất hay khởi nguồn lác đác từ \textbf{viền ngoài phân phối} $\to$ Rất dễ hụt hơi, bỏ lỡ nảy sinh bất thường.
			\end{itemize}

			\vspace{0.5em}
			\begin{exampleblock}{\small Ví dụ: Giao loa phóng thanh cho người vùng thưa thớt}
				\small
				\textbf{Phòng họp đông đúc (trung tâm):} Quá ồn, ai ho 1 tiếng (MMD nhỏ) cũng điếc.\\
				\textbf{Sáng kiến IDW-MMD:} Cấp phép cho "loa phóng thanh" ($w_i \uparrow$) tới tận tay những quan sát thưa thớt ở vùng rìa ngoài (biên). Bất kỳ người lạ mặt (Drift) ở biên xuất hiện, hệ thống sẽ phóng đại rầm rộ ngay lập tức!
			\end{exampleblock}
		\end{column}
		\begin{column}{0.48\textwidth}
			\textbf{Giải pháp IDW-MMD:} Gán trọng số $w_i$ tỷ lệ nghịch với \textit{mật độ cục bộ} $d_i$.

			\begin{block}{Tái cấu trúc Công thức Tính MMD}
				\small
				\begin{equation*}
					w_i \propto \frac{1}{\sqrt{d_i} + \epsilon} \quad \text{với } d_i = \sum_{j} k(x_i, x_j)
				\end{equation*}

				\begin{align*}
					\widehat{\text{MMD}}^2_{\text{IDW}} & = \sum_{i,j} \textcolor{alertred}{w_i w_j} k(x_i, x_j) \\
					                                    & + \sum_{p,q} \textcolor{alertred}{v_p v_q} k(y_p, y_q) \\
					                                    & - \frac{2}{nm} \sum_{i,p} k(x_i, y_p)
				\end{align*}
			\end{block}

			\vspace{0.3em}
			\textbf{Sự thanh lịch toán học:} Chuyển từ duyệt trâu (Permutation Test với độ phức tạp $O(N_{perm} \cdot n^2)$) sang tính \textbf{Asymptotic p-value} qua xấp xỉ phân phối tiệm cận Gamma ($O(n^2)$). Chỉ tính giải eigenvalue 1 lần rưỡi thay vì hoán vị trộn mảng tới 2,500 lần!
		\end{column}
	\end{columns}
\end{frame}

% --- Slide 10: IDW-MMD Calculation Example ---
\begin{frame}{Ví dụ tính toán trực quan: Vì sao IDW "nhạy bén" hơn?}
	\textbf{Giả định tình huống:} Có 3 người bước vào phòng (3 điểm dữ liệu).
    \begin{itemize}
        \item $x_1, x_2$ đứng rất sát nhau ở giữa phòng $\to$ Mật độ $d$ lớn.
        \item $x_3$ đứng lủi thủi 1 mình tự kỷ sát góc khuất $\to$ Mật độ $d$ nhỏ.
    \end{itemize}

	\vspace{1em}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Bước 1: Tính mật độ cục bộ $d_i = \sum k(x_i, x_j)$}
            \begin{itemize}
                \item $d_1 = k(x_1, x_1) + k(x_1, x_2) \approx 1 + 0.9 = \textbf{1.9}$
                \item $d_2 = k(x_2, x_1) + k(x_2, x_2) \approx 0.9 + 1 = \textbf{1.9}$
                \item $d_3 = k(x_3, x_3) + \text{tiếng vọng xa xôi} \approx \textbf{1.0}$
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Bước 2: Cấp quyền ưu tiên $w_i = \frac{1}{\sqrt{d_i}}$}
            \begin{itemize}
                \item $w_1 = w_2 = \frac{1}{\sqrt{1.9}} \approx \textbf{0.72}$ $\to$ Bị bóp nhỏ tiếng lại.
                \item $w_3 = \frac{1}{\sqrt{1.0}} = \textbf{1.0}$ $\to$ To mồm! Ưu tiên tuyệt đối!
            \end{itemize}
        \end{column}
    \end{columns}

	\vspace{1em}
	\begin{alertblock}{\small Vậy bài học rút ra là gì?}
	    \small Người lạ đứng sát mép tường ($x_3$) có tiếng nói "to còi" trong phương trình chênh lệch trọng số ($\textbf{1.0} \gg \textbf{0.72}$). Thuật toán này bẻ cong cái uy quyền đám đông để khơi dậy bất thường. Dịch nôm na: \textit{Triệt tiêu tiếng ồn của đám mây trung tâm, làm căng rõ mồn một các kẻ xâm nhập biên!}
	\end{alertblock}
\end{frame}

\begin{frame}{Đóng góp 2: Ý tưởng cốt lõi thuật toán SE-CDT}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{Làm sao phân loại mà không cần mở Nhãn lên xem?}
			
			Thay vì kiểm tra độ rớt % Accuracy \%%, nghiên cứu này tiên phong tiếp cận phân tích qua \textbf{đường cong hình học (Shape curve)} bằng 9 "chỉ số máu":

			\vspace{0.5em}
			\begin{itemize}
				\item \textbf{Khối Hình Học (Phanh Khui Cấu Trúc):}
				    \begin{itemize}
    				    \item $n_p$: Đếm tổng số nhịp đập Peak.
    				    \item $\text{WR}$: Đo khe hở chóp có gầy không (Width Ratio).
    				    \item $\text{SNR}$: So đo độ nổi loạn của cục Peak với vùng Tạp Âm.
    				\end{itemize}
				\item \textbf{Khối Thời Gian (Đo Nhịp Hành Vị):}
				    \begin{itemize}
				        \item $\text{LTS}$: Số % thời gian đồ thị nằm phía trên đỉnh dốc.
				    \end{itemize}
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{\small Trực quan: Bác sĩ xem điện tâm đồ bắt mạch}
				\small
				Ta không cần thọc kim vô máy lấy máu mới biết bệnh! Ta chỉ nhìn \textbf{biểu đồ điện tim}:
				
				- \textbf{Bệnh Sudden:} Điện tim đâm 1 gai chĩa chóp nhọn thẳng góc ($n_p$ cục bộ, WR cực nhỏ). \textit{Sốc!} \\
				- \textbf{Bệnh Gradual:} Tín hiệu cứ rề rề nhô cao như con rùa leo núi mỏi mệt. \textit{Tăng xông máu!}\\
				- \textbf{Bệnh Blip:} Xe tạt giật đùng đùng rồi tắt điện! $n_p=2$ nảy liền lấp lánh (1 đỉnh dương, 1 đỉnh âm lân cận).
			\end{exampleblock}

% --- Slide 11: Hybrid Design ---
\begin{frame}{Thiết kế lai: IDW-MMD cho Detection, Standard MMD cho Classification}
	\begin{columns}[T]
		\begin{column}{0.52\textwidth}
			\textbf{Trực giác đằng sau thiết kế: Tách rời (Decoupled)}\\
            Tại sao lại thiết kế 2 tầng với 2 loại hàm MMD khác nhau?

			\vspace{0.5em}
			\begin{itemize}
				\item \textbf{Std MMD dễ bị "Báo động giả" (FP):} Do trọng số đều, nó thường xuyên giật mình với các nhiễu cục bộ thông thường $\to$ Rất dở để làm Gác cửa (Detection).
				\item \textbf{IDW-MMD lại bị mù "Gradual Drift":} Việc giảm trọng số vùng trung tâm khiến IDW san phẳng các biến đổi chậm (Gradual/Incremental), biến tín hiệu đồi núi thành đường bằng phẳng $\to$ Không thể dùng để Phân tích (Classification).
			\end{itemize}

			\vspace{0.5em}
			\textbf{Quyết định (Trade-off):} Thiết kế \textit{Một cái chỉ làm nhiệm vụ chặn bắt kẻ lạ, một cái chỉ làm nhiệm vụ phân tích hành vi!}
		\end{column}
		\begin{column}{0.45\textwidth}
			\begin{block}{Quyết định thiết kế}
				\begin{itemize}
					\item \textbf{Tầng Detection:} \textcolor{univblue}{IDW-MMD}\\
					      {\small Tối ưu độ nhạy biên, giảm FP}
					\item \textbf{Tầng Classification:} \textcolor{univblue}{Std MMD}\\
					      {\small Giữ lại thông tin toàn cục}
				\end{itemize}
			\end{block}

			\vspace{0.3em}
			\begin{exampleblock}{\small Bí quyết: Dùng đúng kính cho đúng việc}
				\small
				\begin{itemize}
					\item \textbf{Detection (IDW-MMD):} Là cái \textbf{Kính Lúp} cực nhạy soi ngay cửa ra vào để chặn kẽ hở biên. Độ zoom lớn nên góc rất hẹp $\to$ Cảnh báo nhạy nhất.
					\item \textbf{Classification (Std MMD):} Cất kính lúp, vác \textbf{Ống Nhòm} ra xem để thấy "toàn cảnh bầu trời" (global info). Phải nhìn toàn cảnh mới biết đối tượng "chạy" (Sudden) hay tà tà "đi bộ" (Gradual) để phân loại!
				\end{itemize}
			\end{exampleblock}
		\end{column}
	\end{columns}
\end{frame}

% --- Backup 8: Permutation vs Asymptotic ---
\begin{frame}[noframenumbering]{Backup: Permutation Test vs Asymptotic p-value}
	\textbf{Câu hỏi:} ``Bỏ Permutation Test --- Asymptotic p-value có chính xác bằng không?''

	\vspace{0.5em}
	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\begin{alertblock}{Permutation Test (ShapeDD gốc)}
				\begin{itemize}
					\item Hoán vị nhãn 2,500 lần
					\item \textbf{Exact p-value}: không giả định phân phối
					\item Chi phí: $O(N_{\text{perm}} \cdot n^2)$
					\item Thực tế: $\sim$6.25 triệu phép tính/window
					\item $\to$ \textbf{$\sim$8,000 mẫu/s}
				\end{itemize}
			\end{alertblock}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{Asymptotic p-value (IDW-MMD)}
				\begin{itemize}
					\item Dựa trên phân phối tiệm cận Gamma
					\item Convergence khi $n \geq 50$ (thỏa mãn: $n = 50$)
					\item Chi phí: $O(n^2)$ (eigenvalue decomposition)
					\item $\to$ \textbf{$\sim$131,000 mẫu/s}
				\end{itemize}
			\end{exampleblock}
		\end{column}
	\end{columns}

	\vspace{0.5em}
	\textbf{Kiểm chứng:} F1 tương đương (0.481 vs ShapeDD gốc), \textbf{FP từ vài đơn vị $\to$ 0} $\to$ IDW-MMD + asymptotic thậm chí \textbf{chính xác hơn} do giảm variance
\end{frame}
