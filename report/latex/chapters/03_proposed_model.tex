\chapter{Mô hình ShapeDD-IDW và Hệ thống SE-CDT}
\label{chap:proposed-model}

\textit{Tóm tắt: Trên cơ sở lý thuyết về ShapeDD và các phương pháp phát hiện concept drift đã trình bày ở Chương~\ref{chap:theoretical-foundation}, chương này trình bày việc tích hợp \textbf{Weighted MMD với phân phối tiệm cận} vào framework ShapeDD nhằm cải thiện hiệu suất tính toán, đồng thời đề xuất phương pháp mới SE-CDT cho phân loại drift không giám sát. Luận văn cũng trình bày kiến trúc hệ thống xử lý luồng dữ liệu. Các đóng góp chính bao gồm: (1) Tích hợp Inverse Density-Weighted MMD (IDW-MMD) với asymptotic p-value để tăng tốc độ xử lý, (2) Xây dựng khung chiến lược thích ứng đa dạng dựa trên loại drift, và (3) Thiết kế kiến trúc hệ thống có khả năng mở rộng và chịu lỗi cao.}

\section{Động lực và tổng quan cách tiếp cận}

Mặc dù ShapeDD gốc thể hiện nhiều ưu điểm về độ chính xác định vị và khả năng khử nhiễu, việc áp dụng phương pháp này trong các hệ thống thời gian thực quy mô lớn gặp phải ba thách thức chính:

\begin{enumerate}
	\item \textbf{Chi phí tính toán cao:} Việc sử dụng kiểm định hoán vị với số lượng lớn (thường là 2500 lần để đảm kết quả có ý nghĩa về mặt thống kê) để ước lượng p-value tạo ra nghẽn cổ chai về hiệu năng, giới hạn tốc độ xử lý.
	\item \textbf{Độ nhạy với tham số:} Hiệu suất của ShapeDD phụ thuộc nhiều vào việc lựa chọn độ rộng hạt nhân (kernel bandwidth) và kích thước cửa sổ, làm giảm khả năng tổng quát hóa trên các luồng dữ liệu có đặc tính drift đa dạng.
	\item \textbf{Thiếu cơ chế thích ứng:} Phát hiện drift chỉ là bước đầu; hệ thống cần một cơ chế để tự động cập nhật mô hình một cách phù hợp với bản chất của sự thay đổi (đột ngột hay dần dần)~\cite{yuan2022advances}.
\end{enumerate}

Để giải quyết các thách thức trên, luận văn xây dựng hệ thống \textbf{SE-CDT}, gồm ba phần chính:
\begin{itemize}
	\item \textbf{Tầng thuật toán:} Cải tiến ShapeDD bằng cách tích hợp IDW-MMD (Inverse Density-Weighted MMD), gọi tắt là ShapeDD-IDW, kết hợp phân phối tiệm cận để giảm phương sai và chi phí tính toán, bỏ luôn permutation test tốn kém.
	\item \textbf{Tầng chiến lược:} Xây dựng khung thích ứng thông minh, dùng kết quả phân loại từ SE-CDT để kích hoạt chiến lược cập nhật mô hình phù hợp (Reset, Incremental, hoặc Reuse)~\cite{gama2014survey, lu2018learning}.
	\item \textbf{Tầng kiến trúc:} Triển khai hệ thống trên nền tảng Apache Kafka, hỗ trợ xử lý luồng dữ liệu phân tán, độ trễ thấp và khả năng chịu lỗi.
\end{itemize}

\section{Cải tiến thuật toán phát hiện concept drift}

\subsection{Inverse Density-Weighted MMD (IDW-MMD): Tối ưu hóa hiệu suất và độ chính xác}
\label{sec:idw-mmd}

\subsubsection{Vấn đề của ước lượng MMD truyền thống}
Trong phương pháp gốc, thống kê MMD được tính toán dựa trên trọng số đều, coi vai trò của mọi điểm dữ liệu trong cửa sổ là như nhau. Nhắc lại từ Chương~\ref{chap:theoretical-foundation}, công thức MMD chuẩn có dạng:
\begin{equation}
	\text{MMD}^2_{\text{uniform}}(X, Y) = \frac{1}{n^2}\sum_{i,j} k(x_i, x_j) + \frac{1}{m^2}\sum_{p,q} k(y_p, y_q) - \frac{2}{nm}\sum_{i,p} k(x_i, y_p)
\end{equation}
trong đó các tổng chạy qua tất cả các cặp chỉ số, kể cả $i=j$. Với kernel RBF, thành phần tự đối sánh $k(x_i, x_i) = 1$, nghĩa là mỗi điểm đều đóng góp một giá trị bằng 1 vào tổng, làm giá trị MMD cao hơn thực tế một chút. Hạn chế này sẽ được khắc phục trong IDW-MMD (Mục~\ref{sec:idw-mmd}) bằng cách loại các cặp điểm trùng nhau khỏi phép tính.

Phương pháp này có hai vấn đề chính:
\begin{enumerate}
	\item \textbf{Kết quả không ổn định:} Khi cửa sổ nhỏ ($n, m < 200$), giá trị MMD dao động nhiều, khó phân biệt drift thật với nhiễu.
	\item \textbf{Chậm:} Cần chạy permutation test khoảng 2500 lần để tính p-value, tốn $O(N_{perm} \cdot n^2)$ thời gian.
\end{enumerate}

Vấn đề cốt lõi: Phương pháp uniform coi tất cả điểm dữ liệu quan trọng như nhau. Nhưng thực tế, drift thường xuất hiện đầu tiên ở \textbf{biên phân phối} (nơi có ít điểm) chứ không phải ở vùng trung tâm đông đúc.

\subsubsection{Ý tưởng cải tiến: Dùng trọng số thay đổi theo vị trí}
Thay vì dùng trọng số đều, luận văn đề xuất gán trọng số theo vị trí của điểm trong phân phối: điểm ở vùng thưa (biên) được trọng số cao hơn, điểm ở vùng đông đúc (trung tâm) được trọng số thấp hơn.

MMD có trọng số có dạng tổng quát:
\begin{equation}
	\text{MMD}^2_{\text{weighted}}(X, Y) = \sum_{i,j} w_i w_j k(x_i, x_j) + \sum_{p,q} v_p v_q k(y_p, y_q) - 2\sum_{i,p} w_i v_p k(x_i, y_p)
\end{equation}
trong đó $w_i$ và $v_p$ là trọng số của từng điểm (thỏa mãn $\sum_i w_i = 1$, $\sum_p v_p = 1$). Câu hỏi đặt ra là: tính $w_i$ như thế nào để phát hiện drift hiệu quả?

\subsubsection{Giải pháp đề xuất: Trọng số nghịch biến mật độ (Inverse Density Weighting)}
Luận văn đề xuất phương pháp \textbf{IDW-MMD} (Inverse Density-Weighted MMD), một cách tiếp cận heuristic mới, lấy động lực từ ý tưởng weighted MMD trong Optimally-Weighted MMD của Bharti et al.~\cite{bharti2023owmmd}.

Phương pháp của Bharti et al. được thiết kế cho bài toán suy luận thống kê, đề xuất trọng số tối ưu phức tạp và không liên quan đến bài toán drift detection. Luận văn chỉ kế thừa ý tưởng ``gán trọng số cho từng điểm dữ liệu trong MMD'' và áp dụng một heuristic đơn giản hơn (trọng số nghịch biến với căn bậc hai của mật độ kernel) để phù hợp với bài toán drift detection.

Cụ thể, trọng số $w_i$ được tính dựa trên mật độ cục bộ (local density), chỉ tính các cặp điểm khác nhau (off-diagonal) để tránh bias tự tương quan:
\begin{equation}
	w_i \propto \frac{1}{\sqrt{\sum_{j \neq i} k(x_i, x_j)} + \epsilon} \quad \text{(Inverse Density Weighting)}
\end{equation}
trong đó $\epsilon$ là một hằng số nhỏ (ví dụ $\epsilon = 0.5$) để đảm bảo tính ổn định số học (numerical stability).

Nói đơn giản: điểm ở vùng trung tâm có nhiều điểm lân cận nên $\sum_{j \neq i} k(x_i, x_j)$ lớn, dẫn đến $w_i$ nhỏ; điểm ở vùng biên có ít lân cận hơn nên $w_i$ lớn hơn, phù hợp với trực giác rằng drift thường xuất hiện đầu tiên ở biên phân phối. Về lý do chọn $1/\sqrt{d_i}$ thay vì $1/d_i$: dùng nghịch đảo thẳng cho trọng số quá lớn ở vùng biên, dễ bị nhiễu ảnh hưởng; căn bậc hai giúp tăng trọng số vừa phải hơn. Thực nghiệm cho thấy $1/\sqrt{d_i}$ cho kết quả ổn định nhất (xem Chương~\ref{chap:experiments}).

Dựa trên những phân tích trên, luận văn đề xuất công thức IDW-MMD hoàn chỉnh như sau. Gọi $W^{XX}_{ij}$ và $W^{YY}_{pq}$ là các ma trận trọng số đã được chuẩn hóa theo U-statistic style (loại trừ đường chéo, sao cho $\sum_{i \neq j} W^{XX}_{ij}=1$ và $\sum_{p \neq q} W^{YY}_{pq}=1$), với $W^{XX}_{ij} \propto \tilde{w}_i \tilde{w}_j$ và $\tilde{w}_i = 1/(\sqrt{d(x_i)}+\epsilon)$, $d(x_i) = \sum_{j \neq i} k(x_i, x_j)$:
\begin{equation}	
	\text{MMD}^2_{\text{IDW}}(X, Y) = \underbrace{\sum_{i \neq j} W^{XX}_{ij}\, k(x_i, x_j)}_{\text{Weighted } XX} + \underbrace{\sum_{p \neq q} W^{YY}_{pq}\, k(y_p, y_q)}_{\text{Weighted } YY} - 2\underbrace{\sum_{i=1}^n \sum_{p=1}^m \frac{1}{nm} k(x_i, y_p)}_{\text{Uniform } XY}
\end{equation}

\textbf{Lý do Cross-term dùng trọng số đều.} Thành phần so sánh chéo giữa hai cửa sổ giữ nguyên trọng số đồng nhất $1/(nm)$ vì đây là số hạng đo sự chồng chéo giữa hai phân phối: nếu gán trọng số IDW cho nó sẽ tạo ra khoảng cách giả ngay cả khi hai cửa sổ có cùng phân phối, chỉ do cấu trúc mật độ cục bộ khác nhau. Nói cách khác, Cross-term đóng vai trò ``neo'' trung lập, còn Weighted XX và YY đóng vai trò ``kính lúp'' khuếch đại tín hiệu biên. Do thiết kế này, $\text{MMD}^2_{\text{IDW}}$ có một bias dương nhỏ ngay cả khi $P = Q$. Đây là đặc điểm đã biết và được xử lý qua asymptotic p-value (Mục~\ref{sec:idw-mmd}), cho phép ngưỡng phát hiện được hiệu chỉnh tự động mà không cần permutation test.

\subsubsection{Tại sao IDW-MMD cải thiện hiệu năng phát hiện?}
Khi dùng MMD gốc, tất cả các cặp điểm đóng góp ngang nhau vào kết quả. Điều này khiến tín hiệu trôi dạt (concept drift signal) dễ bị ``chìm'' vào lượng lớn các điểm dữ liệu không thay đổi ở trung tâm phân phối. IDW-MMD khắc phục bằng hai cơ chế: (1) giảm trọng số các điểm trung tâm đông đúc, tránh để chúng ``lấn át'' tín hiệu drift; (2) tăng trọng số các điểm vùng biên, nơi sudden drift thường biểu hiện đầu tiên qua sự xuất hiện đột ngột của phân phối mới. Nhờ vậy IDW-MMD nhạy hơn với các thay đổi rõ ràng ở biên phân phối. Tuy nhiên, chính cơ chế này cũng làm IDW-MMD ít nhạy hơn với gradual drift (thay đổi dần từ vùng trung tâm). Hạn chế này được thảo luận ở cuối mục.

Luận văn trình bày IDW-MMD theo thuật toán cụ thể như sau. Lưu ý rằng mật độ cục bộ (local density) được tính độc lập trong từng cửa sổ để phản ánh đúng cấu trúc nội tại của phân phối đó trước khi so sánh chéo:

\begin{algorithm}[H]
	\caption{Inverse Density-Weighted MMD (IDW-MMD)}
	\label{alg:idw_mmd}
	\begin{algorithmic}[1]
		\Require Reference window $X = \{x_1, \ldots, x_n\}$, test window $Y = \{y_1, \ldots, y_m\}$, kernel $k(\cdot,\cdot)$, stability constant $\epsilon = 0.5$
		\Ensure IDW-MMD statistic $\widehat{\text{MMD}}^2_{\text{IDW}}$
		
		\For{$i = 1$ \textbf{to} $n$} \Comment{inverse density weights for reference window $X$}
			\State $d(x_i) \leftarrow \sum_{j \neq i} k(x_i, x_j)$ \Comment{off-diagonal density; excludes self-kernel $k(x_i,x_i)$}
			\State $\tilde{w}_i \leftarrow 1 \,/\, (\sqrt{d(x_i)} + \epsilon)$
		\EndFor
		\State $\widetilde{W}^{XX}_{ij} \leftarrow \tilde{w}_i\, \tilde{w}_j$ for $i \neq j$, else $0$
		\State $W^{XX}_{ij} \leftarrow \widetilde{W}^{XX}_{ij} \;/\; {\textstyle\sum_{i \neq j} \widetilde{W}^{XX}_{ij}}$ \Comment{normalize so $\sum_{i,j} W^{XX}_{ij} = 1$}
		
		\For{$p = 1$ \textbf{to} $m$} \Comment{inverse density weights for test window $Y$}
			\State $d(y_p) \leftarrow \sum_{q \neq p} k(y_p, y_q)$
			\State $\tilde{v}_p \leftarrow 1 \,/\, (\sqrt{d(y_p)} + \epsilon)$
		\EndFor
		\State $\widetilde{W}^{YY}_{pq} \leftarrow \tilde{v}_p\, \tilde{v}_q$ for $p \neq q$, else $0$
		\State $W^{YY}_{pq} \leftarrow \widetilde{W}^{YY}_{pq} \;/\; {\textstyle\sum_{p \neq q} \widetilde{W}^{YY}_{pq}}$
		
		\State $\widehat{\text{MMD}}^2_{\text{IDW}} \leftarrow
			{\displaystyle\sum_{i,j}} W^{XX}_{ij}\, k(x_i, x_j)
			+ {\displaystyle\sum_{p,q}} W^{YY}_{pq}\, k(y_p, y_q)
			- \frac{2}{nm} {\displaystyle\sum_{i=1}^{n}\sum_{p=1}^{m}} k(x_i, y_p)$
		\Comment{cross-term uses uniform weights $1/(nm)$ as a geometric anchor; diagonal excluded from $XX$, $YY$ terms (U-statistic style)}
		\Return $\widehat{\text{MMD}}^2_{\text{IDW}}$
	\end{algorithmic}
\end{algorithm}

\subsubsection{Phân tích độ phức tạp tính toán}

Độ phức tạp của IDW-MMD được phân tích như sau:

\begin{table}[H]
	\centering
	\caption{Độ phức tạp tính toán của IDW-MMD và các phương pháp liên quan}
	\label{tab:complexity}
	\begin{tabular}{|l|c|c|l|}
		\hline
		\textbf{Thành phần}     & \textbf{Thời gian}   & \textbf{Bộ nhớ} & \textbf{Ghi chú}    \\
		\hline
		Kernel Matrix           & $O(n^2)$             & $O(n^2)$        & $n$ = window size   \\
		Density weights         & $O(n)$               & $O(n)$          & Sum over rows       \\
		Weighted MMD            & $O(n^2)$             & $O(1)$          & Scalar output       \\
		Asymptotic p-value      & $O(n^2)$             & $O(n)$          & Variance estimation trên cross-kernel \\
		\hline
		\textbf{IDW-MMD + Asymptotic} & $O(n^2)$        & $O(n^2)$        & Không cần permutation \\
		\hline
		ShapeDD gốc (2500 perm) & $O(n^2 \times 2500)$ & $O(n^2)$        & Permutation test    \\
		\hline
	\end{tabular}
\end{table}

IDW-MMD có độ phức tạp $O(n^2)$, tương đương MMD chuẩn, nhưng loại bỏ được chi phí permutation test ($O(n^2 \times N_{perm})$) nhờ sử dụng phân phối tiệm cận để tính p-value.

\textbf{Tính p-value không cần permutation.}
Ý tưởng cốt lõi: thay vì xáo trộn dữ liệu 2500 lần để ước lượng ``nếu không có drift thì MMD² thường có giá trị bao nhiêu'', ta tính điều đó \textit{trực tiếp} từ lý thuyết thống kê.

MMD² thực chất là \textit{trung bình} của rất nhiều đóng góp từng cặp điểm. Khi không có drift ($H_0: P = Q$), theo Định lý Giới hạn Trung tâm, trung bình đó hội tụ về phân phối chuẩn khi cỡ mẫu lớn~\cite{gretton2012kernel}:
\begin{equation}
	\sqrt{n}\cdot\text{MMD}^2_u \ \xrightarrow{d}\ \mathcal{N}(0,\,\sigma^2_0)
	\quad\text{khi }H_0\text{ đúng.}
\end{equation}

Thay vì tính đầy đủ $\sigma^2_0$ theo công thức lý thuyết (đòi hỏi duyệt $O(n^2)$ cặp $h(z_i,z_j)$), luận văn dùng xấp xỉ leading-order được đề xuất trong~\cite{gretton2012kernel}: phương sai của MMD² chủ yếu do sự biến thiên của cross-kernel $K_{XY}$ chi phối:
\begin{equation}
	\hat{\sigma}^2_0 \approx \frac{4\,\mathrm{Var}(K_{XY})}{\min(n,m)},
\end{equation}
trong đó $\mathrm{Var}(K_{XY})$ là phương sai mẫu của ma trận $n \times m$ phần tử $k(x_i, y_j)$ — đã có sẵn sau bước tính kernel. Từ đó z-score và p-value (một đuôi, vì MMD² luôn dương) được tính:
\begin{equation}
	z = \frac{\text{MMD}^2_u}{\hat{\sigma}_0}, \qquad
	\text{p-value} = 1 - \Phi(z),
\end{equation}
trong đó $\Phi$ là hàm phân phối tích lũy chuẩn tắc. Nếu p-value $< \alpha$ (thường $\alpha = 0.05$), kết luận có drift.

\textbf{Tại sao hiệu quả?} $\mathrm{Var}(K_{XY})$ chỉ tốn $O(nm)$, không cần thêm vòng lặp nào so với bước tính kernel. Ngược lại, permutation test phải tính lại kernel $N_{perm}=2500$ lần, tổng cộng $O(2500 \times n^2)$. Với cửa sổ $n=50$, thời gian xử lý giảm từ khoảng $125$ms xuống khoảng $7.6$ms mỗi window, nhanh hơn $16.5$ lần.

\subsubsection{Hạn chế và khi nào nên dùng}
Mặc dù IDW-MMD cải thiện đáng kể về tốc độ và độ ổn định, phương pháp có hai giới hạn cần lưu ý:
\begin{enumerate}
	\item \textbf{Không nhạy với mild drift:} Vì giảm trọng số vùng trung tâm, IDW-MMD có xu hướng bỏ qua các thay đổi nhỏ xảy ra ở đó.
	\item \textbf{Không phù hợp để phân loại loại drift:} Cơ chế triệt nhiễu có thể làm mờ các đặc trưng hình dạng cần thiết để phân biệt gradual/incremental drift. Do đó, SE-CDT (Mục~\ref{sec:shaped-cdt}) phân tích \textit{hình dạng} tín hiệu MMD thay vì giá trị tuyệt đối, và cách này về cơ bản độc lập với cơ chế weighting ở bước detection.
\end{enumerate}

Sau khi tích hợp IDW-MMD với phân phối tiệm cận, hệ thống có thể phát hiện drift với độ chính xác cao và chi phí tính toán thấp. Tuy nhiên phát hiện drift chỉ là bước đầu. Để chọn chiến lược thích ứng phù hợp, hệ thống cần biết loại drift đã xảy ra. Phần tiếp theo trình bày phương pháp SE-CDT cho bài toán phân loại drift không giám sát.

\section{Phương pháp phân loại drift SE-CDT}
\label{sec:shaped-cdt}

CDT-MSW là nguồn cảm hứng chính cho SE-CDT, nhưng phương pháp gốc~\cite{guo2022cdtmsw} có một hạn chế lớn. Cả ba tiến trình (phát hiện drift, ước lượng độ dài drift, xác định loại drift) đều dựa trên accuracy ratio $\tilde{P} = a_B / a_A$, tức là so sánh độ chính xác phân loại của cửa sổ test $W_B$ với cửa sổ tham chiếu $W_A$ trên dữ liệu có nhãn. Điều này không khả thi trong môi trường không giám sát.

Luận văn đề xuất \textbf{SE-CDT} (ShapeDD-Enhanced Concept Drift Type), thay thế cả hai tiến trình Growth và Tracking của CDT-MSW bằng cách phân tích hình dạng của tín hiệu MMD $\sigma(t)$ do ShapeDD-IDW tạo ra: dùng FWHM của đỉnh tín hiệu để ước lượng độ dài drift (không cần nhãn), và dùng các đặc trưng hình học lẫn thời gian của tín hiệu để xác định loại drift. Cách tiếp cận này chỉ phân tích hình dạng signal nên về cơ bản độc lập với cơ chế weighting ở bước detection.

\subsection{Cơ sở lý thuyết}
Theo phân loại của CDT-MSW, drift chia thành hai loại chính:
\begin{itemize}
	\item \textbf{TCD (Transient Concept Drift):} Drift tạm thời với thời gian chuyển đổi ngắn, gồm sudden, blip và recurrent (do tính chất lặp lại của các trạng thái ổn định).
	\item \textbf{PCD (Progressive Concept Drift):} Drift tiến triển với thời gian chuyển đổi dài, gồm gradual và incremental.
\end{itemize}

\textbf{Giả định cốt lõi và phạm vi áp dụng.}
Cần làm rõ một điểm về bản chất phương pháp. Các khái niệm ``sudden'', ``gradual'', ``incremental'', ``recurrent'' trong lý thuyết concept drift (Chương~\ref{chap:theoretical-foundation}) được định nghĩa dựa trên cách phân phối liên hợp $P(X, Y)$ thay đổi theo thời gian, bao gồm cả $P(Y|X)$ (ranh giới quyết định). Trong khi đó, SE-CDT là phương pháp không giám sát, chỉ quan sát được sự thay đổi của $P(X)$ qua tín hiệu MMD $\sigma(t)$.

Vì vậy SE-CDT thực chất phân loại \textbf{hình dạng của sự thay đổi phân phối đầu vào} (covariate shift pattern), không phải loại concept drift theo nghĩa supervised truyền thống. Cách tiếp cận này dựa trên \textbf{giả định joint drift}: trong nhiều ứng dụng thực tế, sự thay đổi $P(X)$ và $P(Y|X)$ có tương quan cao, khi phân phối đầu vào dịch chuyển đột ngột thì ranh giới quyết định cũng thường thay đổi đột ngột, và ngược lại~\cite{lu2018learning}. Giả định này được kiểm chứng ở Chương~\ref{chap:experiments} bằng cách dùng các bộ sinh dữ liệu kết hợp (joint drift generators) tạo ra sự thay đổi đồng thời trên $P(X)$ và $P(Y|X)$.

\textit{Khi giả định joint drift không thỏa mãn} (ví dụ Virtual Drift thuần, $P(Y|X)$ thay đổi nhưng $P(X)$ giữ nguyên), SE-CDT sẽ không phát hiện và không phân loại. Đây là hành vi mong đợi, không phải lỗi, vì hệ thống được thiết kế chỉ phản ứng với các thay đổi quan sát được từ dữ liệu không nhãn.

\subsection{Đặc trưng phân loại từ tín hiệu \texorpdfstring{$\sigma(t)$}{sigma(t)}}
SE-CDT trích xuất hai nhóm đặc trưng từ tín hiệu MMD. \textbf{Nhóm 1 --- Đặc trưng hình học} gồm:
\begin{enumerate}
	\item \textbf{Peak Width Ratio:} $\text{WR} = \frac{\text{FWHM}}{2l_1}$, trong đó FWHM (Full Width at Half Maximum) là độ rộng của đỉnh MMD đo tại nửa chiều cao cực đại. WR đo xem đỉnh drift chiếm bao nhiêu phần khoảng quan sát: đỉnh hẹp (WR nhỏ) tương ứng với drift đột ngột, đỉnh rộng (WR lớn) tương ứng với drift kéo dài. Ngưỡng WR $< 0.12$ dùng để loại nhanh TCD/PCD trong bước ước lượng độ dài drift; ngưỡng WR $< 0.15$ kết hợp SNR $> 2.0$ dùng để xác nhận Sudden ở bước phân loại subcategory (xem Algorithm~\ref{alg:se-cdt}).
	\item \textbf{Peak Count ($n_p$):} Số lượng peaks. Sudden có ít peaks ($\leq 3$), Recurrent có nhiều peaks đều đặn ($\geq 4$).
	\item \textbf{Periodicity CV:} Hệ số biến thiên khoảng cách giữa peaks. Recurrent có CV $< 0.3$ (đều đặn).
	\item \textbf{Signal-to-Noise Ratio:} $\text{SNR} = \frac{\max(\sigma)}{\text{median}(\sigma)}$ --- TCD có SNR cao ($> 2.0$, peaks sắc nét).
	\item \textbf{Peak Proximity Ratio (PPR):} Tỉ lệ khoảng cách gần nhất giữa hai peaks so với độ dài tín hiệu. Blip drift có PPR $< 0.20$.
	\item \textbf{Dual-Peak Amplitude Ratio (DPAR):} Tỉ lệ chiều cao giữa hai peaks gần nhau. Blip drift có DPAR $> 0.65$ (hai peaks cao tương đương).
\end{enumerate}

\textbf{Nhóm 2 --- Đặc trưng thời gian} gồm:
\begin{enumerate}
	\item \textbf{Linear Trend Strength (LTS):} Hệ số $R^2$ của hồi quy tuyến tính trên tín hiệu. Incremental có LTS cao ($> 0.5$).
	\item \textbf{Step Detection Score (SDS):} Tỉ lệ các bước nhảy có ý nghĩa ($> 1.5\sigma$) trong tín hiệu.
	\item \textbf{Monotonicity Score (MS):} Tỉ lệ chiều hướng thống trị (tăng hoặc giảm) trong tín hiệu.
\end{enumerate}

\subsection{Thuật toán phân loại}

Quy trình phân loại của SE-CDT được mô tả trong Algorithm~\ref{alg:se-cdt}:

\begin{algorithm}[H]
	\caption{SE-CDT: Unsupervised Drift Type Classification}
	\label{alg:se-cdt}
	\begin{algorithmic}[1]
		\Require Drift magnitude signal $\sigma(t)$ from ShapeDD-IDW
		\Ensure Drift type (TCD or PCD) and subcategory
		\State $\sigma_s \leftarrow \text{GaussianFilter}(\sigma,\; \sigma_g = 4)$ \Comment{smooth the signal}
		\State $P \leftarrow \text{FindPeaks}\!\left(\sigma_s,\; \tau = \bar{\sigma} + 0.3\,\sigma_{\text{std}}\right)$
		\State Compute geometric features: $n_p \leftarrow |P|$, WR, CV, SNR, PPR, DPAR
		\State Compute temporal features: LTS, SDS, MS
		\Statex
		\State $\ell \leftarrow \text{EstimateDriftDuration}(\sigma_s)$ \Comment{ước lượng qua FWHM của đỉnh MMD; thay thế Growth Process của CDT-MSW (vốn dùng accuracy variance có nhãn)}
		\If{$\ell > 1$} \Comment{đỉnh rộng $\Rightarrow$ PCD, phân loại subcategory ngay theo đặc trưng thời gian}
			\If{$\text{LTS} > 0.5$ \textbf{or} ($\text{MS} > 0.6$ \textbf{and} $\text{LTS} > 0.3$) \textbf{or} ($\text{SDS} > 0.12$ \textbf{and} $\text{LTS} > 0.3$) \textbf{or} $n_p \geq 7$}
				\Return \textbf{PCD-Incremental} \Comment{xu hướng tuyến tính rõ hoặc dạng bậc thang}
			\Else
				\Return \textbf{PCD-Gradual} \Comment{drift chậm, mượt mà, không có xu hướng rõ}
			\EndIf
		\EndIf
		\Statex
		\If{$n_p = 2$ \textbf{and} $\text{PPR} < 0.20$ \textbf{and} $\text{DPAR} > 0.65$} \Comment{Step 1 — Blip}
			\Return \textbf{TCD-Blip}
		\EndIf
		\If{$n_p \leq 3$ \textbf{and} $\text{WR} < 0.15$ \textbf{and} $\text{SNR} > 2.0$} \Comment{Step 2 — Sudden}
			\Return \textbf{TCD-Sudden}
		\EndIf
		\If{$n_p \geq 4$ \textbf{and} $\text{CV} < 0.3$ \textbf{and} $\text{LTS} < 0.5$} \Comment{Step 3 — Recurrent}
			\Return \textbf{TCD-Recurrent}
		\EndIf
		\Statex
		\If{$\text{LTS} > 0.5$ \textbf{or} ($\text{MS} > 0.6$ \textbf{and} $\text{LTS} > 0.3$) \textbf{or} ($\text{SDS} > 0.12$ \textbf{and} $\text{LTS} > 0.3$) \textbf{or} $n_p \geq 7$} \Comment{Step 4 — fallback PCD (khi không khớp TCD nào ở trên)}
			\Return \textbf{PCD-Incremental} \Comment{xu hướng tuyến tính rõ hoặc dạng bậc thang}
		\Else
			\Return \textbf{PCD-Gradual} \Comment{drift chậm, là trường hợp mặc định}
		\EndIf
	\end{algorithmic}
\end{algorithm}

\subsection{Ưu điểm và độ phức tạp}

SE-CDT có một số ưu điểm nổi bật so với CDT-MSW:
\begin{itemize}
	\item \textbf{Không giám sát:} Hoạt động theo kiểu không giám sát, không cần nhãn dữ liệu, phù hợp với streaming data.
	\item \textbf{Tích hợp:} Sử dụng cùng tín hiệu IDW-MMD $\sigma(t)$ đã tính từ bước detection, không cần tính toán thêm.
	\item \textbf{Thời gian thực:} Phân loại ngay sau khi phát hiện drift với chi phí $O(m)$ tuyến tính --- rất nhỏ so với bước detection.
	\item \textbf{Phân tách detection và classification:} Detection dùng IDW-MMD (nhạy với sudden/boundary drift); classification phân tích hình dạng của signal đó (geometric + temporal features) để xác định loại drift.
\end{itemize}

Về độ phức tạp tính toán của toàn pipeline:
\begin{itemize}
	\item \textbf{Detection (ShapeDD-IDW):} $O(n^2 \cdot T/\text{step})$ với $n$ = window size, $T$ = độ dài stream, step = bước trượt. Đây là bước chiếm chi phí chính.
	\item \textbf{Classification (SE-CDT)}, chạy sau khi detection trả về $\sigma(t)$ với $m$ điểm:
	\begin{itemize}
		\item Gaussian smoothing: $O(m)$
		\item Peak detection: $O(m)$ --- scipy.signal.find\_peaks
		\item Feature extraction: $O(n_p)$ với $n_p$ = số peaks
		\item Classification: $O(1)$ --- so sánh ngưỡng
	\end{itemize}
	\item \textbf{Tổng classification:} $O(m)$ --- tuyến tính, không đáng kể so với detection.
	\item \textbf{Tổng pipeline:} Bottleneck nằm ở detection: $O(n^2 \cdot T/\text{step})$.
\end{itemize}

\subsection{Giải thích lựa chọn ngưỡng}
Các ngưỡng trong Algorithm~\ref{alg:se-cdt} được xác định qua phân tích thực nghiệm trên synthetic datasets (với cửa sổ chuẩn $n, m \approx 200$) và tinh chỉnh qua nhiều vòng lặp:
\begin{itemize}
	\item \textbf{Blip:} $n_p = 2$, PPR $< 0.20$, DPAR $> 0.65$. Blip drift tạo hai peaks gần nhau (PPR thấp) với biên độ tương đương (DPAR cao), phản ánh mẫu ``drift rồi quay lại''.
	\item \textbf{Sudden:} $n_p \leq 3$, WR $< 0.15$, SNR $> 2.0$. Sudden drift tạo đỉnh sắc nét, hẹp (WR $< 0.15$, tức FWHM chiếm dưới 30\% khoảng quan sát) với tín hiệu nổi rõ trên nền nhiễu (SNR $> 2$). Lưu ý rằng ngưỡng WR $< 0.12$ ở bước Growth Process chặt hơn, nên mọi trường hợp TCD được xác nhận ở đó đều tự động thỏa WR $< 0.15$ ở bước này.
	\item \textbf{Recurrent:} $n_p \geq 4$, CV $< 0.3$, LTS $< 0.5$. Nhiều peaks đều đặn (CV thấp) mà \textit{không} có xu hướng tuyến tính (LTS thấp, để tránh nhầm với Incremental).
	\item \textbf{Incremental vs Gradual:} Phân biệt bằng đặc trưng thời gian. Incremental có xu hướng tuyến tính mạnh (LTS $> 0.5$) hoặc tính đơn điệu cao (MS $> 0.6$). Gradual có tín hiệu dao động (LTS, MS thấp).
\end{itemize}

Các ngưỡng trên được tối ưu cho cửa sổ chuẩn $n, m \approx 200$. Nếu thay đổi kích thước cửa sổ, WR, SNR, PPR cần được hiệu chỉnh lại; với dữ liệu thực tế, khuyến nghị tinh chỉnh (hyperparameter tuning) trên một tập dữ liệu domain-specific nhỏ.

\textbf{Lưu ý:} Các ngưỡng cố định trên mang tính \textit{heuristic}, được xác định qua quan sát thực nghiệm chứ không suy ra từ phân tích thống kê lý thuyết. Để giảm bớt sự phụ thuộc vào con số cứng và giúp SE-CDT thích nghi giữa các luồng dữ liệu khác nhau \textit{mà vẫn không cần nhãn}, luận văn bổ sung thêm một cơ chế \textbf{tự hiệu chỉnh ngưỡng theo baseline} (Mục~\ref{sec:se-cdt-self-calibration}). Luận văn vẫn frame phần Classification này như một \textbf{proof-of-concept}; các hạn chế còn lại và hướng cải tiến được thảo luận trong Chương~\ref{chap:conclusion}.

\subsection{Tự hiệu chỉnh ngưỡng phân loại (Online Self-Calibration)}
\label{sec:se-cdt-self-calibration}

Một điểm yếu cố hữu của bộ ngưỡng cứng là: cùng một giá trị SNR $> 2.0$ có thể vừa đủ ``sắc nét'' đối với một stream Gaussian sạch, nhưng lại quá khó đạt trên một stream có nhiễu nền cao. Để giải quyết vấn đề này theo hướng \textit{vẫn không cần nhãn} (runtime-unsupervised), luận văn bổ sung một mô-đun \textbf{Rolling Feature Baseline} chạy song song với detector.

\textbf{Ý tưởng.} Mỗi khi cửa sổ hiện tại được kết luận là \textit{không có drift}, ba đặc trưng then chốt --- SNR, WR, và CV --- được trích xuất từ tín hiệu MMD và đưa vào một bộ đệm vòng kích thước $B$ (mặc định $B = 200$). Với bộ đệm này, hệ thống ước lượng các phân vị (quantile) thực nghiệm của từng đặc trưng trong điều kiện \textit{không có drift}. Khi cần phân loại một drift mới phát hiện, SE-CDT \textit{nới lỏng} ngưỡng cứng theo phân vị quan sát được trên baseline:
\begin{equation}
	\tau^{\text{adapt}}_{\text{SNR}} = \max\bigl(\tau^{\text{static}}_{\text{SNR}},\; q_{0.90}(\text{SNR baseline})\bigr),
\end{equation}
và tương tự cho WR và CV (với phân vị thấp $q_{0.10}$ thay cho $q_{0.90}$ khi điều kiện là ``$<$''). Lý do chọn cơ chế ``\textit{nới lỏng một chiều}'' (one-sided loosening) thay vì \textit{thay thế hoàn toàn} là để bảo toàn hành vi đã được kiểm chứng trên benchmark synthetic: nếu baseline rất sạch (quantile thấp), ngưỡng cứng tiếp tục được dùng; chỉ khi baseline cho thấy nhiễu nền cao bất thường, ngưỡng mới được kéo lên cho phù hợp.

\textbf{Quy tắc kích hoạt.} Baseline chỉ được cập nhật khi (i) cửa sổ vừa rồi không phát hiện drift và (ii) tín hiệu MMD trace có độ dài tối thiểu (mặc định $\geq 3$ điểm) để đảm bảo SNR/WR/CV có ý nghĩa thống kê. Bộ đệm có \textit{cửa sổ làm nóng} (warm-up): trước khi tích lũy đủ $B_{\min} = 30$ mẫu, hệ thống vẫn dùng ngưỡng cứng và bỏ qua mọi sự nới lỏng. Cơ chế này tránh việc adaptive threshold ``chạy theo'' nhiễu trong giai đoạn đầu của stream.

\textbf{Ưu điểm chính.}
\begin{enumerate}
	\item \textbf{Vẫn không cần nhãn} ở runtime: baseline chỉ dùng tín hiệu MMD, không cần $y$.
	\item \textbf{Không phá vỡ benchmark synthetic:} cơ chế ``one-sided loosening'' đảm bảo trên các stream sạch, ngưỡng vẫn giống cấu hình tĩnh; sự khác biệt chỉ xuất hiện khi nhiễu nền tăng.
	\item \textbf{Tính generic giữa các domain:} cùng một SE-CDT có thể chạy trên dữ liệu công nghiệp nhiễu cao và dữ liệu Gaussian tổng hợp mà không cần tinh chỉnh ngưỡng thủ công cho từng stream.
\end{enumerate}

\subsection{Bộ nhớ khái niệm cho Recurrent Drift (Concept Memory)}
\label{sec:se-cdt-concept-memory}

Trong phiên bản đầu tiên của SE-CDT, các điểm Recurrent drift bị phân loại như một chuỗi sự kiện Sudden riêng lẻ (oracle SUB Recurrent $= 0\%$ trong báo cáo trước cải tiến) vì decision tree không có cách nào ``ghi nhớ'' rằng cùng một concept đã xuất hiện trước đó. Để khắc phục, luận văn bổ sung một mô-đun \textbf{Concept Memory} dạng ring buffer.

\textbf{Cấu trúc.} Memory chứa tối đa $M$ snapshot (mặc định $M = 8$), mỗi snapshot là một mẫu $S_k$ kích thước $n_s$ điểm (mặc định $n_s = 200$) lấy ngay sau khi một drift được phát hiện và \textit{ổn định} trở lại. Cùng với mỗi $S_k$, hệ thống lưu cả median-heuristic gamma $\gamma_k$ tại thời điểm chụp.

\textbf{Quy trình tra cứu.} Khi SE-CDT phát hiện một drift mới:
\begin{enumerate}
	\item Trích xuất snapshot post-drift $S^{\text{new}}$ từ cửa sổ trượt hiện tại.
	\item Với mỗi $S_k \in \text{Memory}$, tính \textit{Standard MMD} giữa $S^{\text{new}}$ và $S_k$ bằng RBF kernel với $\gamma$ là trung bình $(\gamma_k + \gamma_{\text{new}})/2$.
	\item Nếu tồn tại $k^\ast$ sao cho $\text{MMD}^2(S^{\text{new}}, S_{k^\ast}) < \tau_{\text{match}}$ (mặc định $\tau_{\text{match}} = 0.01$), drift được \textit{relabel} thành \textbf{TCD-Recurrent} với chỉ số concept $k^\ast$.
	\item Nếu không match được, $S^{\text{new}}$ được \textit{thêm vào memory} và đẩy snapshot cũ nhất ra (chính sách FIFO).
\end{enumerate}

\textbf{Vì sao dùng Standard MMD thay vì IDW-MMD trong concept memory?} Bài toán so khớp concept ở đây không phải là phát hiện sự thay đổi tinh vi ở biên, mà là \textit{trả lời câu hỏi nhị phân}: hai mẫu này có cùng nguồn gốc phân phối hay không? Standard MMD (unbiased U-statistic) cho ước lượng không thiên lệch của khoảng cách phân phối và có ngưỡng diễn giải dễ hơn theo lý thuyết, do đó phù hợp hơn cho phép so sánh ``ghi nhớ vs. mới''. IDW-MMD vẫn được dùng ở bước detection để giữ tốc độ.

\textbf{Tích hợp vào Algorithm~\ref{alg:se-cdt}.} Concept Memory chạy như một \textit{post-processor} của decision tree: khi nhánh ban đầu trả về một subcategory bất kỳ trong nhóm TCD (Sudden, Blip, Recurrent), bước Concept Memory được gọi và có quyền \textit{override} kết quả thành Recurrent nếu match thành công. Nhóm PCD (Gradual, Incremental) không kích hoạt Concept Memory vì bản chất tích lũy của chúng không phù hợp với khái niệm ``concept lặp lại đột ngột''.

\textbf{Lưu ý về IDW-MMD trong classification:}
Luận văn cũng thử nghiệm dùng trực tiếp IDW-MMD signal (thay vì standard MMD) làm đầu vào cho SE-CDT. Kết quả cho thấy IDW-MMD không phù hợp ở đây vì cơ chế triệt nhiễu của nó cũng vô tình làm mờ các đặc trưng hình dạng drift cần thiết cho phân loại. IDW-MMD hiệu quả ở bước \textit{phát hiện} (có drift hay không), còn SE-CDT dùng standard MMD signal để \textit{phân loại} loại drift. Chi tiết thực nghiệm được trình bày ở Chương~\ref{chap:experiments}.

Kết quả phân loại từ SE-CDT được dùng để kích hoạt chiến lược thích ứng phù hợp, trình bày trong phần tiếp theo.

\section{Khung chiến lược thích ứng mô hình}
\label{sec:model-adaptation-framework}


Dựa trên kết quả phân loại drift từ SE-CDT, hệ thống có thể kích hoạt chiến lược cập nhật mô hình phù hợp. Ý tưởng thiết kế chiến lược thích ứng theo loại drift được trình bày dưới đây lấy cảm hứng từ khung phân loại drift của Guo \textit{et al.}~\cite{guo2022cdtmsw} và các chiến lược thích ứng tổng hợp trong các khảo sát của Gama \textit{et al.}~\cite{gama2014survey}, Lu \textit{et al.}~\cite{lu2018learning}, và Yuan \textit{et al.}~\cite{yuan2022advances}.

\subsection{Triển khai các chiến lược}

\subsubsection{Chiến lược cho Sudden Drift (Trọng tâm nghiên cứu)}
Trong phạm vi luận văn, chiến lược cho \textit{Sudden Drift} được cài đặt và đánh giá chi tiết nhất. Quy trình cụ thể như sau:
\begin{enumerate}
	\item \textbf{Kích hoạt:} Khi SE-CDT phân loại drift là ``Sudden''.
	\item \textbf{Thu thập dữ liệu:} Hệ thống chờ và thu thập một lượng dữ liệu mới ($N_{new}$) ngay sau điểm drift.
	\item \textbf{Huấn luyện lại:} Một instance mô hình mới (ví dụ: Logistic Regression) được khởi tạo với trọng số ngẫu nhiên và huấn luyện trên $N_{new}$.
	\item \textbf{Thay thế:} Mô hình cũ bị loại bỏ hoàn toàn, mô hình mới được đưa vào sử dụng (hot-swap).
\end{enumerate}

\subsubsection{Cơ chế Model Caching cho Recurrent Drift}
Để xử lý drift lặp lại, hệ thống duy trì một \textit{Model Repository}. Mỗi khi một mô hình bị thay thế (do drift), nó được lưu lại cùng với "chữ ký" phân phối (distribution signature) của dữ liệu mà nó đã học.
\begin{itemize}
	\item \textbf{Signature:} Vector trung bình và ma trận hiệp phương sai của đặc trưng.
	\item \textbf{Matching:} Khi drift xảy ra, hệ thống so sánh phân phối dữ liệu mới với các signature trong kho bằng khoảng cách Kolmogorov-Smirnov (KS). Nếu độ tương đồng cao (khoảng cách nhỏ hơn ngưỡng $\epsilon$), mô hình cũ sẽ được tải lại và tinh chỉnh nhẹ.
\end{itemize}

\section{Kiến trúc đề xuất cho hệ thống phát hiện trôi dạt và cập nhật mô hình thích ứng}

Để hiện thực hóa các phương pháp trên trong môi trường thực tế, luận văn đề xuất một kiến trúc xử lý luồng dữ liệu phân tán dựa trên Apache Kafka. Kiến trúc này được thiết kế để đảm bảo tính tách biệt (decoupling), khả năng mở rộng (scalability) và độ tin cậy (reliability).

\begin{quotation}
\textbf{Lưu ý về phạm vi triển khai:} Phần này trình bày \textbf{thiết kế kiến trúc đề xuất}, không phải triển khai production hoàn chỉnh. Cụ thể:
\begin{itemize}
	\item \textbf{Đã triển khai:} Mô phỏng streaming qua Kafka-compatible API (Redpanda) với các thuật toán detection và classification.
	\item \textbf{Chưa đánh giá:} Hiệu năng hệ thống phân tán (latency dưới tải cao, throughput với nhiều partitions, fault recovery time).
\end{itemize}
Đánh giá benchmark trong Chương~\ref{chap:experiments} tập trung vào hiệu năng thuật toán (detection accuracy, runtime), không phải infrastructure performance.
\end{quotation}

	\item Dữ liệu được serialize dưới dạng JSON bao gồm timestamp, feature vector và metadata.
	\item Hỗ trợ điều chỉnh tốc độ phát tin (throughput control) để mô phỏng các tải hệ thống khác nhau.

\subsubsection{Consumer và Cơ chế Windowing}
Thành phần Consumer đóng vai trò quan trọng nhất, thực hiện phát hiện drift:
\begin{itemize}

	\item \textbf{4-Phase Lifecycle:}
	      \begin{enumerate}
		      \item \textit{Pre-training:} Thu thập dữ liệu ban đầu để huấn luyện mô hình gốc.
		      \item \textit{Warm-up:} Giai đoạn ổn định, thiết lập baseline hiệu suất.
		      \item \textit{Monitoring (Frozen):} Mô hình chạy ở chế độ dự đoán (không học), liên tục kiểm tra drift bằng ShapeDD.
		      \item \textit{Adaptation:} Tạm dừng monitoring để cập nhật mô hình khi có drift.

\section{Cơ chế chịu lỗi và đảm bảo độ tin cậy}

Trong môi trường phân tán, các sự cố là không thể tránh khỏi. Hệ thống đề xuất tích hợp các cơ chế chịu lỗi (Fault Tolerance) ở nhiều cấp độ:

\subsection{Xử lý lỗi ở mức ứng dụng}
\begin{itemize}
	\item \textbf{Graceful Degradation:} Nếu module Adaptor gặp lỗi (ví dụ: tràn bộ nhớ khi huấn luyện), hệ thống sẽ fallback về mô hình cũ và ghi log cảnh báo, thay vì dừng toàn bộ pipeline.
	\item \textbf{Try-Catch Wrapper:} Quá trình tính toán ShapeDD được bọc trong các khối an toàn để xử lý các ngoại lệ số học (như ma trận không nghịch đảo) mà không làm crash Consumer.
\end{itemize}

\subsection{Xử lý lỗi ở mức hạ tầng (Kafka)}
\begin{itemize}
	\item \textbf{Replication:} Các topic quan trọng được cấu hình \texttt{replication.factor=3} để đảm bảo không mất dữ liệu ngay cả khi 1-2 broker bị lỗi.
	\item \textbf{Offset Management:} Consumer chỉ commit offset sau khi đã xử lý xong batch dữ liệu, đảm bảo ngữ nghĩa "at-least-once" (không bỏ sót dữ liệu).
	\item \textbf{Consumer Rebalancing:} Nếu một instance của Consumer bị lỗi, Kafka sẽ tự động phân phối lại các partition cho các consumer còn lại trong nhóm.
\end{itemize}

\section{Kết luận chương}
Chương này đã trình bày chi tiết mô hình đề xuất \textbf{SE-CDT}, bao gồm cải tiến thuật toán (IDW-MMD với phân phối tiệm cận), chiến lược thích ứng thông minh, và kiến trúc hệ thống dựa trên Kafka. Cải tiến thuật toán được đánh giá thực nghiệm chi tiết trong Chương~\ref{chap:experiments}, trong khi kiến trúc Kafka được trình bày như một thiết kế đề xuất cho triển khai production trong tương lai.
